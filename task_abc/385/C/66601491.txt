using namespace std;
# define rep(i,n) for (int i = 0; i < (n); i++)
// const int inf = 1073741823;
// const long long INF = 1LL<<60;
int main() {
    int n;
    cin>>n;
    vector<int> h(3500);
    rep(i,n) cin>>h[i];
    int ans = 1;
    for(int d=1;d<n;d++)
    {
        for(int i=0;i<d;i++)
        {
            int  cnt = 0, prv = -1;
            for(int j=i;j<n;j+=d)
            {
                if(prv!=h[j])
                {
                    cnt = 1;
                    prv = h[j];
                }
                else
                {
                    cnt++;
                }
                ans = max(ans,cnt);
            }
        }
    }
    cout << ans << endl;
    return 0;
}
