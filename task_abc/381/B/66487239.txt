#include <bits/stdc++.h>
using namespace std;

int main() {
string s;
cin>>s;
if(s.size()%2!=0){
    cout<<"No"<<endl;
    return 0;
    }
for(int i=0;i<s.length();i+=2){
    if(s[i]!=s[i+1]){
    cout<<"No"<<endl;
    return 0;
    }
}

map<char,int> cnt;
for(char c:s){
    cnt[c]++;
}
for(auto const&[key,val]:cnt){
    if(val!=2){
        cout<<"No"<<endl;
        return 0;
    }
}
cout<<"Yes"<<endl;
return 0;
}
