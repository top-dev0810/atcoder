#include<bits/stdc++.h>
using namespace std;
int s=0,n,b[200100],a[200100][22],f[(1<<20)+100];
int main()
{
    cin>>n;
    for(int i=1;i<=n;i++)
    {
        cin>>b[i];
    }
    for(int i=n;i>0;i--)
    {
        for(int j=0;j<=20;j++)
        {
            a[i][j]=a[i+1][j];
        }
        a[i][b[i]]=i;
    }
    memset(f,0x3f,sizeof(f));
    f[0]=0;
    for(int i=0,z=0;i<(1<<20);i++)
    {
        z=0;
        if(f[i]<=n)
        {
            for(int j=1;j<=20;j++)
            {
                if(!(i&(1<<(j-1))))
                {
                    int x=a[f[i]+1][j];
                    if(x)
                    {
                        x=a[x+1][j];
                        if(x)
                        {
                            f[i|(1<<(j-1))]=min(x,f[i|(1<<(j-1))]);
                        }
                    }
                }
                else
                {
                    z++;
                }
            }
            s=max(z,s);
        }
    }
    cout<<2*s;
    return 0;
}
