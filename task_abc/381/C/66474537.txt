#include <bits/stdc++.h>
using namespace std;
int main(void){
    int n,ans=1;
    cin>>n;
    
    string t;
    cin>>t;
    
    for(int i=0;i<n;i++){
        if(t[i]=='/'){
            int m=1;
            
            while(true){
                int left=i-m,right=i+m;
                if(left<0 or n<=right) break;
                if(t[left]!='1' or t[right]!='2') break;
                m++;
            }
            
            ans=max(ans,2*(m-1)+1);
        }
    }
    
    cout<<ans;
}
