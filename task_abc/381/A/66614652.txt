#include <bits/stdc++.h>
using namespace std;

int main()
{
  int n;
  string s;
  cin >> n >> s;
  
  bool judge = true;
  
  if (n % 2 != 1) judge = false;
  
  int p = (n + 1) / 2;
  
  if (s.at(p - 1) == '/')
  {
    for (int i = 0; i < p - 1; i++)
    {
      if (s.at(i) != '1') judge = false;
    }
    
    for (int i = n - 1; i > p - 1; i--)
    {
      if (s.at(i) != '2') judge = false;
    }
  }
  else judge = false;
  
  if (judge) cout << "Yes";
  else       cout << " No";
}
