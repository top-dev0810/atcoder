#include<iostream>
using namespace std;
int main()
{
    int n;
    string s;
    cin >> n >> s;
    int a=(n+1)/2-1;
    if(n % 2 == 0){
        cout << "No";
        return 0;
    }
    for(int i = 0;i < a;i++){
        if(s[i] != '1'){
            cout << "No";
            return 0;
        }
    }
    if(s[a] != '/'){
        cout << "No";
        return 0;
    }
    for (int i = a+1; i < n; i ++ ){
        if(s[i] != '2'){
            cout << "No";
            return 0;
        }
    }
    cout << "Yes";
    return 0;
}
