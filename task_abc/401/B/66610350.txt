#include<bits/stdc++.h>
using namespace std;
int main(){
  int n;
  cin>>n;
  int sum=0;
  int flag=0;
  for(int i=0;i<n;i++){
    string s;
    cin>>s;
    if(s=="login"){
      flag=1;
    }
    if(s=="logout"){
      flag=0;
    }
    if(s=="private"){
      if(flag==0){
        sum++;
      }
    }
  }
  cout<<sum;
}
