#include <bits/stdc++.h>
using namespace std;

int main() {
  int N;
  cin >> N;
  vector<string> S(N);
  for(int i = 0;i < N;i++) cin >> S[i];
  
  string status = "logout";
  
  int error = 0;
  for(int i = 0;i < N;i++) {
    if(S[i] == "login") status = "login";
    if(S[i] == "logout") status = "logout";
    if(S[i] == "private" && status == "logout") error++;
  }
  cout << error << endl;
  return 0;
}
