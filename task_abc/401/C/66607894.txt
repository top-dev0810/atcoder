#include<bits/stdc++.h>
#define ll long long
using namespace std;
const ll mod=1e9;
ll a[1000007];
int n,k;

int main(){
  cin>>n>>k;
  if(n<k) return cout<<1,0;
  for(int i=0;i<k;i++){
    a[i]=1;
  }
  a[k]=k;
  for(int i=k+1;i<=n;i++){
    a[i]=((a[i-1]+a[i-1])%mod-a[i-k-1]+mod)%mod;
  }
  cout<<a[n];
  return 0;
}
