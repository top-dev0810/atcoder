#include <bits/stdc++.h>
#include <iostream>

using namespace std;

// Constant for 1e9
const long long MOD = 1e9;

// array to hold sum
long long window[1000000];

int customSequence(int n, int k)
{

    if (n < k)
    {
        return 1;
    }

    // for all integers i less then k will be equal to 1.
    for (int i = 0; i < k; i++)
    {
        window[i] = 1;
    }

    window[k] = k;

    // Preforms the calculation: A(i) ​=A(i−K​) + A(i−K+1) ​+ ... + A(i−1)​ for K ≤ i.
    for (int i = k + 1; i <= n; i++)
    {
        window[i] = ((window[i - 1] + window[i - 1]) % MOD - window[i - k - 1] + MOD) % MOD;
    }

    // return the final value.
    return window[n];
}

int main()
{
    int N, K;
    cin >> N >> K;

    cout << customSequence(N, K) << endl;

    return 0;
}
