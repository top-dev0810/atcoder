#include <bits/stdc++.h>

#define sz(x) (int)(x).size()
using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N, M;
    cin >> N >> M;
    vector<int> F(M + 1, 0);
    vector<int> A(N);
    set<int> T;
    for (int i = 0; i < N; i++) {
        cin >> A[i];
        F[A[i]]++;
        T.insert(A[i]);
    }
    int cur = sz(T);
    if (cur < M) {
        cout << "0\n";
        return 0;
    }
    int result = 0;
    for (int i = N - 1; cur == M && i >= 0; i--) {
        F[A[i]]--;
        result++;
        if (F[A[i]] == 0) {
            cur--;
        }
    }
    cout << result << "\n";

    return 0;
}
