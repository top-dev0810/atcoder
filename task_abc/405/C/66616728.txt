typedef long long ll;
typedef unsigned long long ull;
#define rep(i, n) for (int i = 0; i < (int)(n); i++)
#define all(a) a.begin(), a.end()
#define rall(a) a.rbegin(), a.rend()

int main(void) {
    int n;
    cin >> n;
    vector<int> a(n);

    ll sum = 0;
    rep(i, n) {
        cin >> a[i];
        sum += a[i];
    }

    ll ans = 0;
    rep(i, n) {
        sum -= a[i];
        ans += sum * a[i];
    }

    cout << ans << endl;

    return 0;
}
