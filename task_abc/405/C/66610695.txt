#include <bits/stdc++.h>

#define sz(x) (int)(x).size()
using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N;
    cin >> N;
    vector<int> A(N);
    for (int i = 0; i < N; i++) {
        cin >> A[i];
    }
    int64_t T = 0, result = 0;
    for (int i = 0; i < N; i++) {
        result += A[i] * T;
        T += A[i];
    }
    cout << result << "\n";

    return 0;
}
