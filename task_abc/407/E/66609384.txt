#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
typedef pair<int, int> pii;
#define ff first
#define ss second
#define SZ(a) ((int)a.size())
#define pb push_back
#define ALL(v) v.begin(), v.end()



int main() {
  cin.tie(0)->sync_with_stdio(0);
  int tt;
  cin >> tt;
  while (tt--) {
    int n;
    cin >> n;
    n *= 2;
    vector<int> A(n);
    for (int i = 0; i < n; i++) {
      cin >> A[i];
    }
    priority_queue<int> pq;
    ll ans = 0;
    for (int i = 0; i < n; i++) {
      pq.push(A[i]);
      if (i % 2 == 0) {
        ans += pq.top();
        pq.pop();
      }
    }
    cout << ans << '\n';
  }
}
