#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < (n); ++i)

int main(void)
{
    int n, d;
    string s;
    cin >> n >> d >> s;
    char c = '.';
    int cnt = 0;
    for (int i = n; i >= 0; i--) {
        if(s[i] == '@') {
            s[i] = c;
            cnt++;
        }
        if (cnt == d) break;
    }
    cout <<s<< endl;
}
