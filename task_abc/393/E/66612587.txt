
#include<bits/stdc++.h>
using namespace std;
const int MAX = 1e6 + 6;
int freq[MAX], cnt[MAX], res[MAX];
int main(){
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    int n, k;
    cin >> n >> k;
    vector<int>v(n);
    for(int i = 0;i < n;++i){
        int a;
        cin >> a;
        freq[a]++;
        v[i] = a;
    }
    for(int i = 1;i < MAX;++i){
        for(int j = i;j < MAX;j += i){
            if(freq[j])cnt[i] += freq[j];
        }
    }
    for(int i = 1;i < MAX;++i){
        for(int j = i;j < MAX;j += i){
            if(cnt[i] >= k)res[j] = i;
        }
    }
    for(int i = 0;i < n;++i){
        cout << res[v[i]] << '\n';
    }
}
