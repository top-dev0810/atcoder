#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
#define all(x) (x).begin(),(x).end()


int main() {
	ios_base::sync_with_stdio(0);
    cin.tie(0);
    string s;
    cin >> s;
    int n = s.size();
    s = " " + s;
    int ans = 0;
    for(int i=1;i<=n;i++) {
        for(int j=i+1;j<=n;j++) {
            for(int k=j+1;k<=n;k++) {
                if(s[i]=='A' && s[j]=='B' && s[k]=='C' && (j-i == k-j)) ans++;
            }
        }
    }
    cout << ans;
}
