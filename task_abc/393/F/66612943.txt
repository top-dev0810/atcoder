#include<bits/stdc++.h>
using namespace std;

int main(){
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    int n, q;
    cin >> n >> q;
    vector<int>v(n);
    for(int i = 0;i < n;++i){
        cin >> v[i];
    }
    vector<vector<pair<int, int>>>quer(n);
    vector<int>ans(q);
    for(int i = 0;i < q;++i){
        int r, x;
        cin >> r >> x;
        r--;
        quer[r].push_back({x, i});
    }
    vector<int>dp(n, 2e9);
    for(int i = 0;i < n;++i){
        auto it = lower_bound(dp.begin(), dp.end(), v[i]);
        *it = v[i];
        for(auto t : quer[i]){
            int x = t.first, indx = t.second;
            ans[indx] = upper_bound(dp.begin(), dp.end(), x) - dp.begin();
        }
    }
    for(int i = 0;i < q;++i){
        cout << ans[i] << '\n';
    }
}
