#include<bits/stdc++.h>
using namespace std;

int main(){
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    int n;
    cin >> n;
    string s;
    cin >> s;
    vector<int>pos;
    for(int i = 0;i < n;++i){
        if(s[i] == '1')pos.push_back(i);
    }
    int med = ((int)pos.size() - 1) / 2, p = pos[med], l = p - med;
    long long cost = 0;
    for(int i = 0;i < (int)pos.size();++i){
        cost += abs(pos[i] - (l + i));
    }
    cout << cost << '\n';
}
