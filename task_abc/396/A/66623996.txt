#include <bits/stdc++.h>
using namespace std;
int main(){
    int n,a[101];
    cin>>n;
    int flg=0;
    for(int i=0;i<n;i++){
        cin>>a[i];
    }
    for(int i=2;i<n;i++){
        if(a[i]==a[i-1]&&a[i-1]==a[i-2]&&a[i]==a[i-2]){
           flg=1; 
        }
    }
    if(flg==1){
        cout<<"Yes"<<endl;
    }
    else{
        cout<<"No"<<endl;
    }
}
