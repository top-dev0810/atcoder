#include <bits/stdc++.h>
using namespace std;
int i,j,k;

int main()
{
  stack<int> s;
  int q;
  cin >> q;
  int x;
  int y;
  for(i=0; i<100; i++)
  {
    s.push(0);
  }
  for(i=0; i<q; i++)
  {
    cin >> y;
    if(y==1)
    {
      cin >> x;
      s.push(x);
    }
    if(y==2)
    {
      cout << s.top() << endl;
      s.pop();
    }
  }
  return 0;
}
