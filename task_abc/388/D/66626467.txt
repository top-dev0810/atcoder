#include <bits/stdc++.h>

using namespace std;

using ll = long long;

constexpr char newl = '\n';

int main() {
    cin.tie(nullptr);
    ios::sync_with_stdio(false);

    int n;
    cin >> n;

    vector<int> a(n);
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }

    vector<int> p(n + 1, 0);
    for (int i = 0; i < n; i++) {
        a[i] += p[i];

        int li = min(i + a[i] + 1, n);
        p[i + 1]++; p[li]--;
        a[i] -= li - (i + 1);
        p[i + 1] += p[i];
    }

    for (int i = 0; i < n; i++) {
        cout << a[i] << " \n"[i + 1 == n];
    }

    return 0;
}
