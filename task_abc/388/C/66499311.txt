#include <bits/stdc++.h>
using namespace std;
int main(){
//	freopen("biscuit.in", "r", stdin);
//	freopen("biscuit.out", "w", stdout);
	int n;
	cin >> n;
	vector<int> a(n + 1);
	for(int i = 1; i <= n; i++){
		cin >> a[i];
	}
	sort(a.begin(), a.end());
	long long ans = 0;
	for(int i = 1; i <= n; i++){
		int ala = a[i] / 2;
		auto it = upper_bound(a.begin(), a.end(), ala);
		int c = it - a.begin();
		ans += c - 1;
	}
	cout << ans << endl;
}
