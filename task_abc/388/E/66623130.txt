//1d vector input
template <typename T>
void input_v1D(vector<T>& v) {
    for (auto& x : v)
        cin >> x;
}

//2d vector input
template <typename T>
void input_v2D(vector<vector<T>> &v,ll n,ll m){
   for(ll i=0;i<n;i++)
      for(ll j=0;j<m;j++)
         cin>>v[i][j];
}

//graph edge input
template<typename T>
void input_g2D(vector<vector<T>> &g,ll n){
   for(ll i=0;i<n;i++){
      ll x,y;
      cin>>x>>y;
      --x,--y;
      g[x].push_back(y);
      g[y].push_back(x);
   }
}

void solve()
{ 

  int ans = 0;
  int n;
  cin>>n;

  vector<int> v(n);
  vector<bool> visited(n,false);
  input_v1D(v);

  int i = 0;
  int j = n/2;

  while(i<n && j<n){

   if(!visited[i] && 1LL*2*v[i] <= 1LL*v[j] ){
      visited[i] = true;
      visited[j] = true;
      ans++;
      i++;

      while(i<n && visited[i])
         i++;
   }
 
   while(i + 1 > j)
    j++;

   j++;


  }

  cout<<ans<<endl;
}

   while(t--){
