#include<bits/stdc++.h>
using namespace std;
#define int long long
signed main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    int n;
    cin>>n;
    vector<int> v(n);
    for(int i=0;i<n;i++){
        cin>>v[i];
    }
    int ans=0;
    int idx=n/2;
    for(int i=0;i<n/2;i++){
        while(idx<n && v[i]*2>v[idx]){
            idx++;
        }
        if(idx==n){
            break;
        }
        ans++;
        idx++;
    }
    cout<<ans<<"\n";
    return 0;
}
