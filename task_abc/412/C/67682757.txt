#include<bits/stdc++.h>
using namespace std;
#define int long long
#define endl '\n'
void solve(){
	int n; cin>>n;
	vector<int> a(n+1);
	for(int i=1;i<=n;i++) cin>>a[i];
	int a1=a[1],an=a[n];
	sort(a.begin()+1,a.end());
	int ans=1, x=2*a1;
	for(int i=1;i<=n;i++){
		if(a[i]==an&&a[i]<=x){
			ans++;
			cout<<ans<<endl;
			return;
		}
		if(a[i]<=x&&(a[i+1]>x||i==n)){
			ans++;
			x=2*a[i];
		}
	}
	cout<<-1<<endl;
}
signed main(){
	ios::sync_with_stdio(false),cin.tie(0),cout.tie(0);
	int _=1;
	cin>>_;
	while(_--){
		solve();
	}
}
