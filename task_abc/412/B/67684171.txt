#include <bits/stdc++.h>
using namespace std;

int main() {
      string s, t;
      cin>>s;
      cin>>t;
      unordered_set<char> st;
      for (int i=0; t[i]; i++) {
            st.insert(t[i]);
      }
      bool flag = true;
      for (int i=1; s[i] && flag; i++) {
            if (s[i]>='A' && s[i]<='Z') {
                  if (st.find(s[i-1])==st.end()) {
                        cout<<"No\n";
                        return 0;
                  }
            }
      }
      if (flag) cout<<"Yes\n";
}
