void solve()
{
    int n,k,ans=0;
    cin>>n>>k;
    vector<int>a(n+1),suf(n+2);
	for(int i=1;i<=n;i++) 
        cin>>a[i];
    for(int i=n;i>=1;i--)
        suf[i]=suf[i+1]^a[i];
    auto dfs=[&](auto &&self,int pos,int num,int sum)->void
    {
        if(sum==0)
        {
            ans=max(ans,num);
            return;
        }
        if(pos+sum-1==n) 
        {
            ans=max(ans,num^suf[pos]);
            return ;
        }
        self(self,pos+1,num,sum);
	    self(self,pos+1,num^a[pos],sum-1);
    };
	dfs(dfs,1,0,k);
	cout<<ans<<endl;

    return ;
}
signed main()
{
    ios::sync_with_stdio(false), cin.tie(0), cout.tie(0);
    int tc=1;
    while(tc--)
    {
        solve();
    }
    return 0;
}
