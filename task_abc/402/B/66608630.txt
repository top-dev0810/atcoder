#include <bits/stdc++.h>
using namespace std;

int main() {
  int N;
  cin >> N;
  queue<int> X;
  
  for(int i = 0; i < N; i++) {
    int q;
    cin >> q;
    if(q == 1) {
      int nowX;
      cin >> nowX;
      X.push(nowX);
    }else {
      cout << X.front() << endl;
      X.pop();
    }
  } 
  return 0;
}
