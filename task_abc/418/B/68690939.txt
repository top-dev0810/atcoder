#include <bits/stdc++.h>
using namespace std;

typedef long long ll;
typedef long double ld;

#define sp <<" "<<
#define endl "\n"

void solve() {
    string s; cin >> s;
    int n = s.size();
    vector<int> pos, pref(n);
    for (int i = 0; i < n; i++) {
        if (i) pref[i] = pref[i-1];
        if (s[i] == 't') {
            pref[i]++;
            pos.push_back(i);
        }
    }

    auto query = [&](int l, int r) -> int {
        return pref[r] - (l ? pref[l-1] : 0);
    };

    ld mx = 0;
    for (auto &i : pos) {
        for (auto &j : pos) {
            if (j <= i) continue;
            mx = max(mx, ld(query(i, j) - 2) / ld(j - i - 1));
        }
    }

    cout << mx << endl;
}

signed main() {
    cin.tie(0);
    ios_base::sync_with_stdio(false);

    cout << fixed << setprecision(20) << endl;

    int t = 1;
    // cin >> t;
    while (t--)
        solve();
}
