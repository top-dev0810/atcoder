#include <bits/stdc++.h>
using namespace std;

int main() 
{
    string s;
    cin >> s;
    int n = s.length();
    double m = 0.0;
    
    for (int i = 0; i < n; i++) 
	{
        for (int j = i + 2; j < n; j++) 
		{
            if (s[i] == 't' && s[j] == 't') 
			{
                int len = j - i + 1;
                int sum = 0;
                for (int k = i; k <= j; k++) 
				{
                    if (s[k] == 't') 
					{
                        sum++;
                    }
                }
                double r = (sum - 2) * 1.0 / (len - 2);
                if (r > m) 
				{
                    m = r;
                }
            }
        }
    }
    
    cout << fixed << setprecision(15) << m << endl;
    return 0;
}
