#include <bits/stdc++.h>

using namespace std;
typedef long long ll;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n, q;
    cin >> n >> q;
    vector<ll> a(n), psum(n + 1);
    for (int i = 0; i < n; ++i) cin >> a[i];
    sort(a.begin(), a.end());
    for (int i = 0; i < n; ++i) psum[i + 1] = psum[i] + a[i];
    while (q--) {
        int B;
        cin >> B;
        if (B > a.back()) {
            cout << -1 << '\n';
            continue;
        }
        auto i = lower_bound(a.begin(), a.end(), B - 1) - a.begin();
        ll ans = psum[i] + (n - i) * (B - 1) + 1;
        cout << ans << '\n';

    }  

    return 0;
}
