#include<bits/stdc++.h>
#define endl '\n'
#define en ' '
#define PLL pair<int,int>
using namespace std;
using LL = long long;
int vis[1007];
int n,m,s,t;
int flag=0;
void dfs(int x,auto& arr,auto& ans)
{
    if(flag==1) return;
    if(x==t)
    {
        ans.push_back(t);
        flag=1;
        return;
    }
    vis[x]=1;
    ans.push_back(x);
    for(int i=0;i<arr[x].size();i++)
    {
        int p=arr[x][i];
        if(vis[p]==0) dfs(p,arr,ans);
        if(flag==1) return;
    }
    if(flag==1) return;    
    ans.pop_back();
}
int main()
{
    ios::sync_with_stdio(0);
    cin.tie(0);
    int T;
    cin>>T;
    while(T--)
    {
        flag=0;
        cin>>n>>m>>s>>t;
        for(int i=1;i<=n;i++) vis[i]=0;
        vector<vector<int>> arr(n+1);
        vector<int> ans;
        while(m--)
        {
            int u,v;
            cin>>u>>v;
            arr[u].push_back(v);
            arr[v].push_back(u);
        }
        for(int i=1;i<=n;i++)
        {
            sort(arr[i].begin(),arr[i].end());
        }
        dfs(s,arr,ans);
        for(int i=0;i<ans.size();i++)
        {
            cout<<ans[i]<<en;
        }
        cout<<endl;
    }
}
