void isac()
{
    int n, q;
    cin >> n >> q;

    multiset <int> s;
    for (int i = 1; i <= n; i ++) {
        int x;
        cin >> x;

        s.insert(x);
    }

    while (q --) {
        int x;
        cin >> x;

        if (s.find(x) != s.end()) {
            s.erase(s.find(x));
        }
    }

    for (auto x : s) {
        cout << x << ' ';
    }
}

{
    ios::sync_with_stdio(0);

    int T = 1;
    // cin >> T;
    for (int i = 1; i <= T; i ++) {
        isac();
    }
    return 0;
}
