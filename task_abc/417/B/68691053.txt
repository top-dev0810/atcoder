#include<bits/stdc++.h>
#define endl '\n'
#define en ' '
#define PLL pair<LL,LL>
using namespace std;
using LL = long long;
int main()
{
    int n,m;
    cin>>n>>m;
    vector<int> a(n+1);
    for(int i=1;i<=n;i++) cin>>a[i];
    map<int,int> mp;
    while(m--)
    {
        int x;
        cin>>x;
        mp[x]++;
    }
    for(int i=1;i<=n;i++)
    {
        if(mp[a[i]])
        {
            mp[a[i]]--;
        }
        else cout<<a[i]<<en;
    }
}
