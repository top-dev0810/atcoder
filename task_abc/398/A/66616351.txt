#include <bits/stdc++.h>
#define rep(i, n) for (int i = 0; i < (n); i++)
using namespace std;

int main(){
    int n;
    cin >> n;
    if(n%2 == 0){
        rep(i, n){
            if(i == n/2 || i == n/2 - 1){
                cout << "=";
            }else{
                cout << "-";
            }
        }
    }else{
        rep(i, n){
            if(i == n/2){
                cout << "=";
            }else{
                cout << "-";
            }
        }
    }
    return 0;
}
