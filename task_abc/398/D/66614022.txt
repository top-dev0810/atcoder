int main() {

  string s;
  CIN(n, r, c, s);
  int x = 0, y = 0;
  set<P> st;
  string ans = "";
  int nx = c, ny = -r;
  rep(i, n) {
    st.emplace(x, y);
    if (s[i] == 'N') {
      y--; ny--;
    }
    if (s[i] == 'S') {
      y++; ny++;
    }
    if (s[i] == 'E') {

    }

    else ans += '0';
  }
