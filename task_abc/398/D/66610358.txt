void samekoSaba(){

    int n, a,b; cin >> n >> a >> b;
    string s; cin >> s;

    int x = 0, y = 0;

    map<pair<int,int>, int> mp;
    mp[{0,0}] = 1;

    for ( int i = 0; i < n; i++ ){
        char c = s[i];

        if ( c == 'N' ){
            x++;
            a++;
        } else if ( c == 'W' ){
            y++;
            b++;
        } else if ( c == 'S' ){
            a--;
            x--;
        } else {
            y--;
            b--;
        }

        mp[{x,y}] = 1;

        if ( mp[{a,b}] > 0 ) cout << 1;
        else cout << 0;
    }

}
