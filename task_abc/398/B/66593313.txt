#include <bits/stdc++.h>
using namespace std;

int main() {
    vector<int> number(14);

    for (int i = 0; i < 7; i++) {
        int tmp;
        cin >> tmp;
        number.at(tmp)++;
    }

    int cnt2 = 0; 
    int cnt3 = 0;
    for (int i = 1; i < 14; i++) {
        if (number.at(i) >= 3) {
            cnt3++;
        }
        if (number.at(i) >= 2) {
            cnt2++;
        }
    
    }

    if (cnt3 >= 1 && cnt2 >= 2) {
        cout << "Yes" << endl;
    } else {
        cout << "No" << endl;
    }


}
