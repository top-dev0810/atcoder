void fokakmneeee() {
    int n;
    cin >> n;
    int a[n];
    for (int i = 0; i < n; i++) cin >> a[i];
    int cnt = 0;
    vector<int> ans;
    for (int i = 1; i < n; i++) {
        if (a[i] > a[i - 1]) {
            cnt++;
        } else if(cnt) {
            ans.push_back(cnt);
            cnt = 0;
        }
    }
    ans.push_back(cnt);
    long long sum = 0;
    for (int i = 1; i < ans.size(); i++) {
        sum += 1ll * ans[i] * ans[i - 1];
    }
    cout << sum;
}

int main() {
    input();
    int t = 1;
    // cin >> t;
    while (t--) {
        fokakmneeee();
        if (t)cout << '\n';
    }

    return 0;
}
