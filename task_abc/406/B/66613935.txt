#include <bits/stdc++.h>
using namespace std;
int n,k;
vector<long long> a;
long long val = 1;
long long _10k;
int main(){
    cin >> n >> k;
    a = vector<long long>(n,0);
    _10k = pow(10,k);
    for(int i = 0; i < n; i++){
        cin >> a[i];
        //val = val * a[i];//桁溢れする
        __int128_t tmp = (__int128_t)val * a[i];
        if((tmp < LLONG_MIN) || (tmp > LLONG_MAX)){
            //オーバーフロー時
            val = 1;
        }else{
            if(tmp >= _10k){
                val = 1;
            }else{
                val = tmp;
            }
        }
        //cout << i << " " << a[i] << " " << val << endl;
    }
    cout << val << endl;
    return 0;
}
