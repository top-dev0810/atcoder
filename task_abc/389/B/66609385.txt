#include <bits/stdc++.h>

using i64 = long long;

void solve()
{
    i64 x;
    std::cin >> x;
    i64 y = 1;
    for (int i = 1; ; i++) {
        y *= i;
        if (y == x) {
            std::cout << i << "\n";
            return;
        }
    }
}

int main()
{
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int _ = 1;
    //std::cin >> _;
    while (_--) {
        solve();
    }

   return 0;
}
