#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
const int MAX=0x7fffffff>>1;
int dp[510][510];
vector <int> st[510];
int cnt[510][510];
int n,m,l;
int main() {
	ios::sync_with_stdio(0);cin.tie(0);cout.tie(0);
	cin>>n>>m>>l;
	int ai;
	for (int i=0;i<n;i++) {
		cin>>ai;
		st[i%l].push_back(ai);
	}
	for (int i=0;i<510;i++) {
		for (int j=0;j<510;j++) {
			dp[i][j]=INT_MAX;
		}
	}
	for (int i=0;i<l;i++) {
		for (int j=0;j<m;j++) {
			for (int k=0;k<(int)st[i].size();k++) {
				if (st[i][k]>j) cnt[i][j]+=m;
				cnt[i][j]+=j-st[i][k];
			}
		}
	}
	//dp[i][j]=min(sum(cnt(0~i)))%m==j,i=0~l-1,j=0~m-1;
	for (int i=0;i<m;i++) dp[0][i]=cnt[0][i];
	for (int i=1;i<l;i++)  {
		for (int j=0;j<m;j++) {
			for (int k=0;k<m;k++) {
				dp[i][(j+k)%m]=min(dp[i-1][k]+cnt[i][j],dp[i][(j+k)%m]);
			}
		}
	}
	cout<<dp[l-1][0];
	return 0;
}
