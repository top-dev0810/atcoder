#include<bits/stdc++.h>
using namespace std;
using ll = long long;
const int M=550;
int g[M][M];
int f[M][M]; 
int a[M];
int main()
{
	int n,m,l;
	cin>>n>>m>>l;
	for(int i=1;i<=n;i++) cin>>a[i];
	for(int i=1;i<=l;i++)
	{
		for(int j=0;j<m;j++)
		{
			for(int k=i;k<=n;k+=l)
			{
				g[i][j]+=(j-a[k]+m)%m;
			}
		}
	}
	memset(f,0x3f,sizeof(f));
	f[0][0]=0;
	for(int i=1;i<=l;i++)
	{
		for(int j=0;j<m;j++)
		{
			for(int k=0;k<m;k++)
			{
				f[i][j]=min(f[i][j],f[i-1][k]+g[i][(j-k+m)%m]);
			}
		}
	}
	cout<<f[l][0]<<endl;
	return 0;
 }
