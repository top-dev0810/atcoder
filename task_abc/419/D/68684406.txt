#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

void solve(){
    int n,m;
    cin>>n>>m;
    string s,t;
    cin>>s>>t;
    vector<int> diff(n+1,0);
    for(int i=0;i<m;i++){
        int l,r;
        cin>>l>>r;
        l--;r--;
        diff[l]++;
        diff[r+1]--;
    }   
    for(int i=1;i<=n;i++) {
        diff[i]+=diff[i-1];
    }
    for(int i=0;i<n;i++){
        if(diff[i]&1){
            swap(s[i],t[i]);
        }
    }
    cout<<s<<endl;
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);

    int test=1;
    //cin>>test;
    while(test--){
        solve();
    }
    return 0;
}
