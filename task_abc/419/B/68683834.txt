#include <bits/stdc++.h>
using namespace std;
int main()
{
    int n, q, a;
    int Q[100] = {0};
    //for(int i : Q) cout<<i;
    cin>>n;
    for(int i = 0; i < n; i++){
        cin>>q;
        if(q == 1){
            cin>>a;
            Q[a-1]++;
        }else{
            for(int i = 0; i < 100; i++){
                if(Q[i] > 0){
                    cout<<i+1<<endl;
                    Q[i]--;
                    break;
                }
            }
        }
    }
}
