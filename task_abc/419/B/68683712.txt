#include <bits/stdc++.h>
using namespace std;

vector <int> nums;

int main() {
    ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);

    int q, ac, x;
    cin >> q;

    while (q--){
        cin >> ac;

        if (ac == 1){
            cin >> x;
            nums.push_back(x);
            
        } else{
            sort(nums.begin(), nums.end());
            cout << nums.front() << '\n';
            nums.erase(nums.begin());

        }
    }
    
    return 0;
}
