#include<bits/stdc++.h>
using namespace std;
int n;

int main() {
    ios :: sync_with_stdio(false);
    cin >> n;
    int lx, rx, ly, ry;
    lx = ly = (int)1e9;
    rx = ry = 0;
    for(int i = 1; i <= n; ++i) {
        int x, y;
        cin >> x >> y;
        lx = min(lx, x), rx = max(rx, x);
        ly = min(ly, y), ry = max(ry, y);
    }
    cout << max((rx - lx - 1) >> 1, (ry - ly - 1) >> 1) + 1;
    return 0;
}
