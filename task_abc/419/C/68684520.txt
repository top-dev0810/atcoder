#include <bits/stdc++.h>
using namespace std;
int main()
{
    int n, r, c;
    pair<int, int> rm, cm;
    cin>>n;
    cin>>r>>c;
    rm = {r, r};
    cm = {c, c};
    for(int i = 0; i < n; i++){
        cin>>r>>c;
        rm.second = max(rm.second, r);
        rm.first = min(rm.first, r);
        cm.second = max(cm.second, c);
        cm.first = min(cm.first, c);
    }
    cout<<max((rm.second - rm.first + 1) / 2, (cm.second - cm.first + 1) / 2);
}
