#include <bits/stdc++.h>

#define itn int
#define ll long long
#define ld long double

using namespace std;



int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);
itn n,q;
    cin>>n>>q;
    map<int,int> mp;
    vector<int> v(q);
    int ans=0;
    for(int i=0;i<q;i++) {
        cin>>v[i];
        int k=v[i];
        mp[k]++;
        if (mp[k]%2==0) {
            if (mp[k-1]%2==0&&mp[k+1]%2==0) ans--;
            else if (mp[k-1]%2!=0&&mp[k+1]%2!=0) ans++;
        }
        else {
            if (mp[k-1]%2==0&&mp[k+1]%2==0) ans++;
            else if (mp[k-1]%2!=0&&mp[k+1]%2!=0) ans--;
        }
        cout<<ans<<"\n";
    }
    return 0;
}
