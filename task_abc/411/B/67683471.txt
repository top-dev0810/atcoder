#include<bits/stdc++.h>
using namespace std;
typedef long long ll;

const ll N = 5e1 + 9;
ll n; 
ll d[N];
ll s[N];

int main() {
    scanf("%lld", &n);
    for (ll i = 1; i < n; i++) {
        scanf("%lld", &d[i]);
        s[i] = s[i - 1] + d[i];
    }
    for (ll i = 1; i < n; i++) {
        for (ll j = i + 1; j <= n; j++) {
            printf("%lld ", s[j - 1] - s[i - 1]);
        }
        printf("\n");
    }
}
