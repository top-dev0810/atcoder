#include <bits/stdc++.h>
using namespace std;
using ll=long long;
using ld=long double;
using vl = vector<long long>;
using vi = vector<int>;
using vvl = vector<vector<long long>>;
using vvi = vector<vector<int>>;
using vs = vector<string>;

int main(){
  int N;
  cin>>N;
  vi P(N),Q(N);
  for(int i=0;i<N;i++){
    cin>>P[i];
    Q[i]=P[i];
  }
  sort(Q.rbegin(),Q.rend());
  map<int,int>mp;
  int r=1;
  mp[Q[0]]=r;
  for(int i=1;i<N;i++){
    if(Q[i]==Q[i-1])r++;
    else{mp[Q[i]]=r+1;r++;}
  }
  for(int i=0;i<N;i++){
    cout<<mp[P[i]]<<endl;
  }
}
