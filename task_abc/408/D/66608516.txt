#include <bits/stdc++.h>
using namespace std;
const int N = 1e6+5;
int pre[N];
void solve(){
	int n;
	cin >> n;
	string s;
	cin >> s;
	s = " " + s;
	for(int i = 1; i <= n; i++) {
		pre[i] = pre[i-1] + s[i] - '0';
	}
	int res = min(pre[n], n - pre[n]);
	int mi = -1;
	for(int r = 1; r <= n; r++){
		res = min(res, pre[n] - 2*pre[r] + r + 1 + mi);
		mi = min(mi, 2*pre[r] - r - 1);
	}
	cout << res << endl;
}

int main(){
	int T;
	cin >> T;
	while(T--) solve();
 	return 0;
}
