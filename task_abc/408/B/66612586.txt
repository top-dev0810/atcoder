#include "bits/stdc++.h"
using namespace  std; 
int a[110]; 
int main(){
    int n;
    cin>>n;
    for(int i=1;i<=n;i++) 
        scanf("%d", &a[i]);
    sort(a+1, a+1+n); 
    int m = unique(a+1, a + 1+n) - a - 1; 
    cout<<m<<endl; 
    for(int i=1;i<=m;i++) printf("%d ", a[i]); 
    return 0; 
}
