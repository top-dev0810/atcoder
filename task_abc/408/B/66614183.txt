#include <bits/stdc++.h> 
using namespace std;  


int main() {
    cin.tie(0);
    ios::sync_with_stdio(false);


    int64_t n;
    cin>>n;

    map<int64_t, int64_t> mp;
    for(int64_t i=0;i<n;i++){
        int64_t wk;
        cin>>wk;

        if(mp.contains(wk)){
            mp[wk]++;
        }else{
            mp[wk] = 1;
        }
    }


    cout<<mp.size()<<endl;
    for(auto p:mp){
        cout<<p.first<<" ";
    }cout<<endl;

    return 0;
}
