#include "bits/stdc++.h"
using namespace std; 
int n, m; 
const int N = 1000010; 
int s[N]; 
int main(){
    cin>>n>>m;
    for(int i=1;i<=m;i++){
        int l, r; 
        scanf("%d%d", &l, &r); 
        s[l] +=1; 
        s[r+1]-=1; 
    }
    for(int i=1;i<=n;i++)
        s[i] += s[i-1]; 
    int ans = m;
    for(int i=1;i<=n;i++)
        ans = min(ans, s[i]);
    printf("%d\n", ans); 
    return 0; 

}
