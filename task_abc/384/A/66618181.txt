#include <bits/stdc++.h>
typedef long long ll;

int main() {
  int n = 0;
  char a, b;
  string s;
  cin >> n >> a >> b >> s;
  
  for (int i = 0; i < n; i++) {
    if (s[i] != a) {
      cout << b;
    }
    else {
      cout << s[i];
    }
  }
  cout << endl;
}
