#include <bits/stdc++.h>
using namespace std;
 
int main() {
  int n = 5;
  vector<int> a(n);
  for(int i = 0; i < n; i++) cin >> a[i];
  
  vector<pair<int, string>> rank;
  for(int i = 1; i < (1<< 5); i++) {
    string name;
    int score = 0;
    for(int j = 0; j < 5; j++) {
      if(i & (1 << j)) {
        score += a[j];
        name += 'A' + j;
      }
    }
    rank.emplace_back(-score, name);
  }
  sort(rank.begin(), rank.end());
  for(auto [score, name] : rank) {
    cout << name << endl;
  }
}
