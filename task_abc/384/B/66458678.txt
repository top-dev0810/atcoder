#include <iostream>
using namespace std;

int main() {
    int N, R;
    cin >> N >> R;

    for (int i = 0; i < N; i++) {
        int division, score;
        cin >> division >> score;

        if (division == 1 && 1600 <= R && R <= 2799)
            R += score;
        if (division == 2 && 1200 <= R && R <= 2399)
            R += score;
    }

    cout << R << endl;
    return 0;
}
