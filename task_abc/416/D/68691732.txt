void solve() {

    ll ans=0;
    for(int i=1;i<=n;i++) {
        cin>>a[i];
        a[i]%=m;
        ans+=a[i];
    }
    for(int i=1;i<=n;i++) {
        cin>>b[i];
        b[i]%=m;
        ans+=b[i];
    }
    sort(a.begin()+1,a.end());
    sort(b.begin()+1,b.end());
    vector<int>vis(n+1,0);

    int l=1;
    for(int i=n;i>=1;i--) {
        for (;l<=n;l++) {
            if (a[i]+b[l]>=m) {
                l++;
                ans-=m;
                break;
            }
        }
    }
    cout<<ans<<'\n';
}
