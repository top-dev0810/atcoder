#include<bits/stdc++.h>
using namespace std;
const int N=100005;
int n,k;
string s[N],ans="";
bool cmp(string a,string b)
{
	return a+b==b+a?a<b:a+b<b+a;
}
int main()
{
	scanf("%d%d",&n,&k);
	for(int i=1;i<=n;i++)
		cin>>s[i];
	sort(s+1,s+1+n,cmp);
	while(k>0&&ans.length()<=30)
	{
		string maxx=s[1]+ans;--k;
		for(int i=2;i<=n;i++)
			maxx=min(maxx,s[i]+ans);
		ans=maxx;
	}
	while(k--) printf("%s",s[1].c_str());
	printf("%s\n",ans.c_str());
	return 0;
}
