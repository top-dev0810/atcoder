#include<bits/stdc++.h>
using namespace std;
const int N = 1e5+5;
int n,k;string s[N],ans;
bool cmp(string s1,string s2){
    return (s1+s2==s2+s1?s1<s2:s1+s2<s2+s1);
}
int main(){
    cin>>n>>k;
    for(int i=1;i<=n;i++)cin>>s[i];
    sort(s+1,s+n+1,cmp);
	  ans="";
    while(k>0 && ans.size()<s[1].size()){
        string tmp=s[1]+ans;k--;
        for(int i=2;i<=n;i++)tmp=min(tmp,s[i]+ans);
		ans=tmp;
    }
    while(k--){
    	cout<<s[1];
    }
	cout<<ans;
    return 0;
}
