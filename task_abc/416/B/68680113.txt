#include<iostream>
#include<string>
using namespace std;

int main()
{
    string s;
    cin >> s;

    string t;
    t = s;
    for(int i=0;i<s.length();i++){
        if(s[i] == '#') t[i] = '#';
        else t[i] = '.';
    }

    int count = 1;

    for(int i=0;i<s.length();i++){
        if(count==1 && t[i] == '.'){
            t[i] = 'o';
            count = 0;
        }
        if(t[i] == '#') count = 1;
    }

    for(int i=0;i<t.length();i++) cout << t[i];
}
