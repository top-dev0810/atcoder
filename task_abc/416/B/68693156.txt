#include <bits/stdc++.h>
using namespace std;
#define rep1(i, n) for(int i = 0; i<(int)n; i++)
#define rep2(i, m, n) for(int i = (int)m; i<(int)n; i++)

int main()
{
  string S; cin >> S;
  int N = S.size();
  
  vector<char> T(N);
  rep1 (i, N) {
    if (S.at(i) == '#') T.at(i) = '#';
  }
  
  if (N > 1) {
    rep2 (i, 1, N) {
      if (T.at(i) == '#') {
        for (int j = i-1; j >= 0; j--) {
          if (T.at(j) != '#') {
            T.at(j) = 'o';
            break;
          }
        }
      }
    }
  }
  
  for (int i = N-1; i >= 0; i--) {
    if (T.at(i) != '#') {
      T.at(i) = 'o';
      break;
    }
  }
  
  rep1 (i, N) {
    if (T.at(i) != 'o' && T.at(i) != '#')  T.at(i) = '.';
    cout << T.at(i);
  }
  cout << endl;
  
  return 0;
}
