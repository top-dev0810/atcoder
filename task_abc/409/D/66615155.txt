#include<bits/stdc++.h>
using namespace std;
#define IOS cin.tie(nullptr)->sync_with_stdio(0),cin.exceptions(cin.failbit);
#define lb(x) (x)&-(x)
#define all(x) (x).begin(),(x).end()
#define ll long long

constexpr int maxN=2e5+5;

int n;

inline void solve(){
    cin>>n;
    string s;cin>>s;
    for(int i = 0;i<n-1;i++)if(s[i]>s[i+1]){
        for(int j = i+1;j<=n;j++)if(j==n||s[j]>s[i]){
            char c = s[i];
            for(int k = i;k<j-1;k++)s[k] = s[k+1];
            s[j-1] = c;
            break;
        }
        break;
    }
    cout<<s<<'\n';
}

int main(){
    IOS
    int t;
    cin>>t;
    for(;t--;)solve();
}
