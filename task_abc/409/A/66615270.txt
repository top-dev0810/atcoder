#include <bits/stdc++.h>
using namespace std;
int main() {
  ios::sync_with_stdio(false);
  cin.tie(nullptr);
  int N;
  cin >> N;
  string T, A;
  cin >> T >> A;
  string ans = "No";
  for (int i = 0; i < N; i++) {
    if (T[i] == 'o' && A[i] == 'o') ans = "Yes";
  }
  cout << ans << '\n';
  return 0;
}
