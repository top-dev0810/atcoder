#include<bits/stdc++.h>
using namespace std;
const int N=2e6+5,M=998244353;
long long F[N],G[N],H[N],g[N],f[N],p1,p2,w[N],ans;int tr[N],m;
long long Pow(long long x,int y)
{
	long long pre=1;
	while(y>0)
	{
		if((y&1)==1)
		pre=pre*x%M;
		x=x*x%M;y=y>>1;
	}
	return pre;
}
void NTT(int op)
{
	int i,j,len,p;long long pre;
	for(i=0;i<m;i++)
	{
		if(i<tr[i])
		swap(f[i],f[tr[i]]);
	}
	w[0]=1;w[1]=Pow(3,(M-1)/m);
	if(op==0)
	w[1]=Pow(w[1],M-2);
	for(i=2;i<m;i++)
	w[i]=w[i-1]*w[1]%M;
	for(len=2,p=1;len<=m;len=len<<1,p=p<<1)
	{
		for(i=0;i<m;i+=len)
		{
			for(j=0;j<p;j++)
			{
				pre=f[i+j+p]*w[m/len*j]%M;
				f[i+j+p]=f[i+j]-pre;
				if(f[i+j+p]<0)
				f[i+j+p]+=M;
				f[i+j]=f[i+j]+pre;
				if(f[i+j]>=M)
				f[i+j]-=M;
			}
		}
	}
}
int main()
{
	int n,i,p;
	scanf("%d%d",&n,&p);
	if(p==100)
	{
		for(i=1;i<=n;i++)
		printf("1\n");
		return 0;
	}
	p1=Pow(100,M-2)*p%M;p2=Pow(100,M-2)*(100-p)%M;
	F[0]=1;
	for(i=1;i<=n;i++)
	F[i]=F[i-1]*i%M;
	G[n]=Pow(F[n],M-2);
	for(i=n;i>=1;i--)
	G[i-1]=G[i]*i%M;
	for(i=1;i<=n;i++)
	H[i]=Pow(i,M-2);
	g[n]=1;
	for(i=n-1;i>=1;i--)
	g[i]=g[i+1]*((p2*H[i]+1)%M)%M;
	ans=g[1];g[1]=0;
	for(i=2;i<=n;i++)
	g[i]=(g[i]*F[i-2]%M)*Pow(p2,i)%M;
	for(i=1;i<=n;i++)
	g[i+n]=g[i],g[i]=0;
	for(i=0;i<=n;i++)
	f[n-i]=G[i];
	m=1;
	while(m<=3*n)
	m=m<<1;
	for(i=1;i<m;i++)
	{
		tr[i]=tr[i>>1]>>1;
		if((i&1)==1)
		tr[i]|=(m>>1);
	}
	NTT(1);
	for(i=0;i<m;i++)
	swap(f[i],g[i]);
	NTT(1);
	for(i=0;i<m;i++)
	f[i]=f[i]*g[i]%M;
	NTT(0);
	long long pre=Pow(m,M-2);
	printf("%lld\n",ans);
	p2=Pow(p2,M-2);
	for(i=2;i<=n;i++)
	{
		ans=f[i+n*2]*pre%M;
		ans=(ans*Pow(p1,i-1)%M)*Pow(p2,i)%M;
		ans=ans*G[i-2]%M;
		printf("%lld\n",ans);
	}
	return 0;
}
