#include <bits/stdc++.h>

using namespace std;
const int MaxN = 3e5 + 5;
long long cnt[MaxN] = {0}, d[MaxN];
int main()
{
    int n, L, nowpos = 0;
    cin >> n >> L;
    if(L%3) {
        cout << 0;
        return 0;
    }
    cnt[0]++;
    for(int i = 1; i < n; i++) {
        cin >> d[i];
        nowpos = (nowpos+d[i]) % L;
        cnt[nowpos]++;
    }
    int len = L/3;
    long long ans = 0;
    for(int i = 0; i < L; i++) {
        ans += cnt[i] * cnt[(i+len)%L] * cnt[(i+len+len)%L];
    }
    cout << ans/3;
    return 0;
}
