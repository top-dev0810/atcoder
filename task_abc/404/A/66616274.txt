#define rep(i, n) for (int i = 0; i < (n); i++)
using namespace std;

int main(){
    string s;
    cin >> s;
    vector<bool> a(26, false);
    rep(i, s.size()){
        a[s[i] - 'a'] = true;
    }
    rep(i, 26){
        if(!a[i]){

            break;
        }
    }
    return 0;
}
