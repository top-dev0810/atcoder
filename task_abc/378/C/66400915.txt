#include <bits/stdc++.h>
using namespace std;
int main() {
  int N;
  cin>>N;
  vector<int> A(N);
  for(int i=0;i<N;i++){
    cin>>A[i];
  }
  map<int,int> mp;
  vector<int> B(N);
  for(int i=0;i<N;i++){
    mp[A[i]]=-1;
  }
  for(int i=0;i<N;i++){
    if(mp[A[i]]==-1){
       B[i]=-1;
    }
    else{
      B[i]=mp[A[i]];
    }
    mp[A[i]]=i+1;
  }
  for(int i=0;i<N;i++){
    cout<<B[i]<<" ";
  }
}
