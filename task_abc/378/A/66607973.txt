#include<bits/stdc++.h>
#define endl '\n'
#define int long long
#define double long double
using namespace std;

void solve(){
	int a,b,c,d,cnt=0,say1=0,say2=0,say3=0,say4=0;
	cin>>a>>b>>c>>d;
	vector<int>v={a, b, c, d};
	for(int i=0; i<4; i++){
		if(v[i]==1) say1++;
		else if(v[i]==2) say2++;
		else if(v[i]==3) say3++;
		else if(v[i]==4) say4++;
	}
	if(say1>=2) cnt+=say1/2;
	if(say2>=2) cnt+=say2/2;
	if(say3>=2) cnt+=say3/2;
	if(say4>=2) cnt+=say4/2;
	cout<<cnt;
} 
signed main (){
    ios_base::sync_with_stdio(0);
    cin.tie(0); cout.tie(0);
    int t=1;
    //cin>>t;
    while (t--){
        solve();
    }
}
