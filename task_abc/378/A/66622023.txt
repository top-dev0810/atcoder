#include<iostream>
using namespace std;
int main()
{
    int a[4],x=0,y=0,m=0,n=0;
    for (int i = 0; i < 4; i ++ ){
        cin >> a[i];
        if(a[i] == 1)x++;
        if(a[i] == 2)y++;
        if(a[i] == 3)m++;
        if(a[i] == 4)n++;
    }
    x /= 2;
    y /= 2;
    m /= 2;
    n /= 2;
    cout << x+y+m+n;
    return 0;
}
