#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define rep(i, n) for (int i = 0; i < n; ++i)

int main()
{
    ll t, n;
    string ss;

    cin >> t;
    rep(_, t)
    {
        cin >> n >> ss;
        vector<int> dp((1 << n));
        dp[0] = 1;

        rep(i, 1<<n)
        {
            if (ss[i-1] == '1') continue;
            rep(k, n)
            {
                if (dp[i & ~(1<<k)])
                {
                    dp[i] = 1;
                    break;
                }    
            }
        }
        cout << ((dp[(1<<n)-1]) ? "Yes\n" : "No\n");
    }
}
