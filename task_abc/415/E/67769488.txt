#include<bits/stdc++.h>
using namespace std;
#define ll long long 
const int mod=1e9+7;
const int N=17600;
void solve(){
    int h,w;
    cin>>h>>w;
    vector<vector<int>>a(h+1,vector<int>(w+1));
    for(int i=1;i<=h;++i){
        for(int j=1;j<=w;++j){
            cin>>a[i][j];
        }
    }
    vector<int>p(h+w+1);
    for(int i=1;i<=h+w-1;++i) cin>>p[i];
    ll l=0,r=2e14;
    auto check=[&](ll mid){
        queue<pair<pair<int,int>,ll>>q;
        vector<vector<ll>>vis(h+1,vector<ll>(w+1,-1));
        vis[1][1]=mid+a[1][1]-p[1];
        if(vis[1][1]<0) return false;
        q.push({{1,1,},vis[1][1]});
        while(!q.empty()){
            auto [u,v]=q.front();
            auto [x,y]=u;
            // if(mid==1){
            //     cerr<<x<<" "<<y<<" "<<v<<'\n';
            // }
            q.pop();
            if(v<vis[x][y]) continue;
            if(x+1<=h&&vis[x+1][y]<v-p[x+y]+a[x+1][y]) q.push({{x+1,y},v-p[x+y]+a[x+1][y]}),vis[x+1][y]=v-p[x+y]+a[x+1][y];
            if(y+1<=w&&vis[x][y+1]<v-p[x+y]+a[x][y+1]) q.push({{x,y+1},v-p[x+y]+a[x][y+1]}),vis[x][y+1]=v-p[x+y]+a[x][y+1];
        }
        return vis[h][w]>-1;
    };
    while(l<=r){
        ll mid=l+r>>1ll;
        // cerr<<mid<<'\n';
        if(check(mid)) r=mid-1;
        else l=mid+1;
    }
    cout<<l<<'\n';
} 
signed main(){
    cin.tie(0)->sync_with_stdio(false);
    int t=1;
    // cin>>t;
    while(t--) solve();
    return 0;
}
