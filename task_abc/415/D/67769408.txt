void fokakmneeee() {
    long long have, n;
    cin >> have >> n;
    vector<tuple<long long, long long, long long> > v(n);
    for (int i = 0; i < n; i++) {
        auto &[a, b, c] = v[i];
        cin >> b >> c;
        a = b - c;
    }
    long long ans = 0;
    sort(v.begin(), v.end());
    for (int i = 0; i < n; i++) {
        auto &[a, b, c] = v[i];
        if (have >= b) {
            long long mx = (have - b) / a + 1;
            ans += mx;
            have -= mx * a;
        }
    }
    cout << ans;
}

int main() {
    input();
    int t = 1;
    // cin >> t;
    while (t--) {
        fokakmneeee();
        if (t) cout << '\n';
    }

    return 0;
}
