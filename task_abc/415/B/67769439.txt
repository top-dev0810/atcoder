#include <bits/stdc++.h>
#define rep(i, n) for(int i = 0; i < (int)(n); i++)
#define all(obj) (obj).begin(), (obj).end()
using ll = long long;
using namespace std;

int main() {
    string s;cin >> s;
    int len = (int)s.length();
    int cnt = 0;
    for(int i = 0; i< len; i++){
        if( s[i] == '#'){
            cnt++;
        }
    }
    for(int i = 0; i < cnt/2; i++){
        for(int j = 0; j < len; j++){
            if( s[j] == '#'){
                cout << j+1 << ",";
                s[j] = '.';
                break;
            }
        }
        for(int j = 0; j < len; j++){
            if( s[j] == '#'){
                cout << j+1 << endl;
                s[j] = '.';
                break;
            }
        }
    }
}
