#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for (int i = 0; i< (n); ++i)
using ll=long long;

int main() {
    string s;
    cin >> s;
    vector<int> ans;
    int l = s.length();
    rep(i,l) { 
        if (s[i]=='#') {
            ans.push_back(i+1);
        }
    }
    int n = ans.size()/2;
    rep(i,n) {
        cout << ans[2*i] << ',' << ans[2*i+1] << endl;
    }
    return 0;
}
