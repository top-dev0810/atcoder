#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for (int i = 0; i< (n); ++i)
using ll=long long;

int main() {
    int n;
    cin >> n;
    vector<int> a(n);
    rep(i,n) {
        cin >> a[i];
    }
    int x;
    cin >> x;
    for(auto t : a){
        if (x==t){
            cout << "Yes" << endl;
            return 0;
        }
    }
    cout << " No" << endl;
    return 0;
}
