#include <bits/stdc++.h>
using namespace std;

void print_vector1i(const vector<int>& x){
    cout << "[";
    for (int i=0; i<x.size(); i++){
        cout << x[i];
        if (i != x.size() - 1) cout << " ";
    }
    cout << "]" << endl;
}

int main() {
    int N, Q; cin >> N >> Q;
    int ans = 0;
    
    vector<int> whereIsHato(N,0);
    for(int i=0; i<N; i++) whereIsHato[i] = i;
    
    vector<int> howManyHato(N,1);
    
    for(int i=0; i<Q; i++){
        int order; cin >> order;
        if(order == 1){
            int P, H; cin >> P >> H;
            P--;
            H--;
            if(howManyHato[whereIsHato[P]] == 2) ans--;
            if(howManyHato[H] == 1) ans++;
            howManyHato[whereIsHato[P]]--;
            whereIsHato[P] = H;
            howManyHato[H]++;
        }else{
            cout << ans << endl;
        }
    }
}
