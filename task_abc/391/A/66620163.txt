#include <bits/stdc++.h>

#include <bits/stdc++.h>
typedef long long ll;

int main() {
  string s;
  cin >> s;
  
  if (s == "N") {
    cout << "S" << endl;
  }
  else if (s == "E") {
    cout << "W" << endl;
  }
  else if (s == "W") {
    cout << "E" << endl;
  }
  else if (s == "S") {
    cout << "N";
  }
  else if (s == "NE") {
    cout << "SW" << endl;
  }
  else if (s == "SW") {
    cout << "NE" << endl;
  }
  else if (s == "NW") {
    cout << "SE" << endl;
  }
  else if (s == "SE") {
    cout << "NW" << endl;
  }
}
