#include <bits/stdc++.h>
using namespace std;
#define all(a) (a).begin(), (a).end()
#define st first
#define se second
typedef long long ll;
typedef long double lb;
typedef pair<int, int> ii;

const int INF = 1e9;
const ll LINF = 1e18;
const int dx[8] = {0, 0, -1, 1, 1, 1, -1, -1};
const int dy[8] = {-1, 1, 0 ,0, -1, 1, -1, 1};

int n, m, k; 
map<ii, bool> cnt, vis;
bool ok;

bool check(int x, int y) {
    return (x >= 1 && x <= n && y >= 1 && y <= m && cnt[{x, y}]);
}

void dfs(int x, int y) {
    // cout << x << ' ' << y << '\n';
    vis[{x, y}] = 1;
    if(x == 1 || y == m) {
        ok = 0;
        return;
    }

    for(int i = 0; i < 8; i++) {
        int nx = x + dx[i];
        int ny = y + dy[i];
        if(check(nx, ny) && !vis[{nx, ny}]) dfs(nx, ny);
    }
}

int main() {
	ios::sync_with_stdio(0);
	cin.tie(0); cout.tie(0);

//	freopen("test.inp","r",stdin);
//	freopen("test.out","w",stdout);

    cin >> n >> m >> k;
    
    vector<ii> s;
    for(int i = 1; i <= k; i++) {
        int x, y; cin >> x >> y;
        cnt[{x, y}] = 1;
        if(x == n || y == 1) s.push_back({x, y});
    }

    ok = 1;
    for(auto [x, y] : s) {
        if(!vis[{x, y}]) dfs(x, y);
    }

    cout << (ok ? "Yes\n" : "No\n");
}
