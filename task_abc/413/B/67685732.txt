#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for(int i = 0; i < (n); ++i)
#define yes cout << "Yes" << endl
#define no cout << "No" << endl
using ll = long long;

int main(){
    int n;
    cin >> n;
    vector<string> x(n);
    rep(i, n) cin >> x[i];

    set<string> ans;
    rep(i, n){
        rep(j, n){
            if(i != j){
                ans.insert(x[i] + x[j]);
            }
        }
    }
    cout << ans.size() << endl;
    return 0;
}
