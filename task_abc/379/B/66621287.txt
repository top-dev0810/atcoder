#include<bits/stdc++.h>
using i64 = long long;
int main(){
  std::ios::sync_with_stdio(0);
  std::cin.tie(nullptr);
  
  int N , K;
  std::cin >> N >> K;
  
  std::string s;
  std::cin >> s;
  
  s += 'X';
  
  int cnt = 0 , ans = 0;
  
  for(auto ch : s){
    if(ch == 'O'){
      cnt++;
    }
    if(ch != 'O'){
      ans += cnt / K;
      cnt = 0;
    }
  }
  
  std::cout << ans << '\n';
}
