void solve() {
    int n;
    string s;
    cin>>n>>s;
    s=' '+s;

    V<int>pre(n+1,0);

    FOR(i,1,n){
        pre[i]=pre[i-1]+(s[i]-'0')*i;
    }
    V<int>ans(1,0);
    FORD(i,2,n){
        ans.pb(pre[i]%10);
        pre[i-1]+=pre[i]/10;
    }
    while(pre[1]){
        ans.pb(pre[1]%10);
        pre[1]/=10;
    }
    FORD(i,1,ans.sz-1)cout<<ans[i];
}

signed main() {
    int Task = 1;

    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    while (Task--) {
        solve();
    }

    return 0;
