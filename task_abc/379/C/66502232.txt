int main(){
    ll N,M;
    cin >> N >> M;
	vector<Pll> P(M);
	rep(i,M) cin >> P[i].fi;
	rep(i,M) P[i].fi--;
	rep(i,M) cin >> P[i].se;
	sort(P.begin(), P.end());
	P.push_back(Pll(N, 0));
	ll prev = 0;
	ll buf = 0;
	ll ans = 0;
	for(auto [x, a]:P){
		ll len = (x - prev);
		if(len > buf) drop(-1);
		ans += (buf - (len))*(len);
		ans += (len)*(len-1)/2;
		buf -= len;
		buf += a;
		prev = x;
	}
	if(buf != 0) drop(-1);
	cout << ans << endl;
}
