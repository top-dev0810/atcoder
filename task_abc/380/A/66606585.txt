#include<iostream>
using namespace std;
int main()
{
    int a,q=0,w=0,e=0;
    cin >> a;
    while(a > 0){
        int b=a % 10;
        if(b == 1)q++;
        if(b == 2)w++;
        if(b == 3)e++;
        a/=10;
    }
    if(q == 1 && w == 2 && e == 3)cout << "Yes";
    else cout << "No";
    return 0;
}
