int dfs(int flag){
	int s=0,p=1;
	for(int i=0;i<tot;i++,p*=3){ //状压1第一位 2第二位 0在桌子上
		if(vis[i]==1) s+=p;
		else if(vis[i]==2) s+=2*p;
	}
	int &res=f[s][flag]; 
	if(res!=-1) return res; //记忆化搜索
	
	if(flag){ //先手
		for(int i=0;i<tot;i++){ //12
			if(vis[i]==1){
				vis[i]=3;
				for(int j=0;j<tot;j++){ //12
					if(vis[j]==3&&a[i]>a[j]){//从桌子上回收一张牌
						vis[j]=1;
						//注意这里要将vis[i] vis[j]回溯
						if(!dfs(flag^1)) return vis[i]=1,vis[j]=3,res=1;
						vis[j]=3; //回溯
					}
				}
				//注意这里要将vis[i]回溯
				if(!dfs(flag^1)) return vis[i]=1,res=1;
				vis[i]=1;
			}
		}
	}
	else{
		for(int i=0;i<tot;i++){
			if(vis[i]==2){
				vis[i]=3;
				for(int j=0;j<tot;j++){
					if(vis[j]==3&&a[i]>a[j]){//从桌子上回收一张牌
						vis[j]=2;
						if(!dfs(flag^1)) return vis[i]=2,vis[j]=3,res=1;
						vis[j]=3; //回溯
					}
				}
				//对手输了，我们这次获胜
				if(!dfs(flag^1)) return vis[i]=2,res=1; //直接放在桌子上
				vis[i]=2;
			}
		}
	}
	return res=0;
}


void solve(){
	cin>>n>>m>>K;
	for(int i=0;i<n;i++,tot++) cin>>a[tot],vis[tot]=1; //第一个人
	for(int i=0;i<m;i++,tot++) cin>>a[tot],vis[tot]=2; //第二个人
	for(int i=0;i<K;i++,tot++) cin>>a[tot],vis[tot]=3; //桌子上的牌
	memset(f,-1,sizeof f);
	cout<<(dfs(1)?"Takahashi":"Aoki")<<"\n";
}
