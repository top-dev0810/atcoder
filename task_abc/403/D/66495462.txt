int main(){
    ll N,D;
    cin >> N >> D;
	vector<ll> A(N);
	rep(i,N) cin >> A[i];
	vector<ll> cnt(1000001);
	rep(i,N) cnt[A[i]]++;
	if(D == 0){
		ll ans = 0;
		rep(i,1000001){
			if(cnt[i]){
				ans += cnt[i] - 1;
			}
		}
		drop(ans);
	}
	ll ans = 0;
	rep(i,D){
		vector<ll> v;
		for(ll j=i; j < 1000001; j+=D){
			v.push_back(cnt[j]);
		}
		ll n = v.size();
		vector<vector<ll>> dp(n+1, vector<ll>(2,1e18));
		dp[0][0] = 0;
		dp[0][1] = 0;
		rep(i,n){
			ll a = v[i];
			cmin(dp[i+1][0], dp[i][0] + a); 
			cmin(dp[i+1][0], dp[i][1] + a);
			cmin(dp[i+1][1], dp[i][0]);
		}
		ans += min(dp[n][0], dp[n][1]);
	}
	cout << ans << endl;
}
