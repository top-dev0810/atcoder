#include<bits/stdc++.h>
using namespace std;
#define int long long
#define psb push_back
const int N=2e4+10;
int n,dp[N][2];
string s[N][2];
signed main(){
	ios::sync_with_stdio(0); cin.tie(0),cout.tie(0);
	memset(dp,0x3f,sizeof dp);
	cin>>n;
	dp[1][0]=dp[1][1]=1,s[1][0]=s[1][1]="1";
	dp[11][0]=dp[11][1]=2,s[11][0]=s[11][1]="11";
	dp[111][0]=dp[111][1]=3,s[111][0]=s[111][1]="111";
	dp[1111][0]=dp[1111][1]=4,s[1111][0]=s[1111][1]="1111";
	for(int i=2;i<=n;++i){
		for(int j=1;j<i;++j)
			if(dp[i][0]>dp[j][0]+dp[i-j][0]+1)
				dp[i][0]=dp[j][0]+dp[i-j][0]+1,s[i][0]=s[j][0]+"+"+s[i-j][0];
		if(dp[i][1]>dp[i][0]+2) dp[i][1]=dp[i][0]+2,s[i][1]="("+s[i][0]+")";
		for(int j=2;j<i;++j)
			if(i%j==0&&dp[i][1]>dp[j][1]+dp[i/j][1]+1)
				dp[i][1]=dp[j][1]+dp[i/j][1]+1,s[i][1]=s[j][1]+"*"+s[i/j][1];
		if(dp[i][0]>dp[i][1]) dp[i][0]=dp[i][1],s[i][0]=s[i][1];
	}
	cout<<s[n][0];
	return 0;
}
