#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, m, q;
    cin >> n >> m >> q;

    vector<bool> isOk(n,false);
    vector<set<int>> user(n);
    for (int qi = 0; qi < q; qi++) {
        int type; cin >> type;
        if (type == 1) {
            int x, y; cin >> x >> y;
            x--; y--;
            user[x].insert(y);
        } else if (type == 2) {
            int x; cin >> x;
            x--;
            isOk[x] = true;
        } else if (type == 3) {
            int x, y; cin >> x >> y;
            x--; y--;
            if (isOk[x] || user[x].find(y) != user[x].end()) {
                cout << "Yes" << endl;
            } else {
                cout << "No" << endl;
            }
        }
    }
    return 0;
}
