#include <iostream>
int main() 
{
    int N;
    std::cin >> N;
    
    int A[100];
    for (int i = 0; i < N; ++i) 
    {
        std::cin >> A[i];
    }
    
    int K;
    std::cin >> K;
    
    int count = 0;
    for (int i = 0; i < N; ++i) {
        if (A[i] >= K) {
            count++;
        }
    }
    std::cout << count << "\n";
}
