#include <bits/stdc++.h>
using namespace std;

int main() {
    int n,m;
    cin >> n >> m;
    vector<vector<pair<int,int>>>G(n);
    for(int i = 0;i < m;i++){
        int a,b,w;
        cin >> a >> b >> w;
        a--,b--;
        G[a].push_back({b,w});
    }

    vector<vector<bool>>seen(n,vector<bool>(2 << 10,false));
    queue<pair<int,int>>Q;
    Q.push({0,0});
    seen[0][0] = true;
    while(!Q.empty()){
        auto [v,w] = Q.front();Q.pop();
        for(auto u : G[v]){
            if(seen[u.first][u.second ^ w] == false){
                Q.push({u.first,u.second ^w});
                seen[u.first][u.second ^ w] = true;
            }
        }
    }

    for(int i = 0;i < (2 << 10);i++){
        if(seen[n-1][i] == true){cout << i << endl;return 0;}
    }
    cout << -1 << endl;
    return 0;
}
