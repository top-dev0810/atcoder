#include <bits/stdc++.h>
using namespace std;

int main() {
    int N, Q;
    cin >> N >> Q;
    vector<int> count(N,0);
    vector<int> answer(Q);
    // for (int x:count) cout << x << " ";
    for (int i =0;i<Q;i++){
        int X;
        cin >> X;
        if (X>=1){
            count[X-1]++;
            answer[i]=X;
        } else{
            auto it = min_element(count.begin(),count.end());
            int index = distance(count.begin(),it);
            (*it)++;
            answer[i]=index+1;
        }
    }
    for (int x:answer){
        cout << x << " ";
    }

    return 0;
}
