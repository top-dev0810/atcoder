#include <bits/stdc++.h>
using namespace std;

#define int long long

signed main() {
    int n, h, m; cin >> n >> h >> m;
    int dp[n + 9][m + 9];
    memset(dp, 0, sizeof(dp));
    // dp[i][j] -> max health tersisa setelah kalahin monster i dengan magic power = j
    // memset(dp, h, sizeof(dp));

    vector<pair<int,int>> monster;
    monster.push_back({0,0});
    
    for(int i = 1; i <= n; i++){
        int x,y; cin >> x >> y;
        monster.push_back({x,y});
    }

    for(int j = 0; j <= m; j++){
        dp[0][j] = h;
    }
    
    for(int i = 1; i <= n; i++){
        int monsterhealth = monster[i].first;
        int monstermana = monster[i].second;
        // cout << "I: " << i << '\n';

        //dp[i][2]
        for(int j = 0; j <= m; j++){
            if(j - monstermana < 0){
                dp[i][j] = dp[i-1][j] - monsterhealth;
            }
            else dp[i][j] = max(dp[i-1][j] - monsterhealth, dp[i-1][j - monstermana]);
            // cout << "     " << dp[i][j] << '\n';
        }
    }


    int mx = 0;
    for(int i = 1; i <= n; i++){
        // cout << dp[i][m] << '\n';
        if(dp[i][m] >= 0){
            mx = i;
        }
    }
    cout << mx;
}
