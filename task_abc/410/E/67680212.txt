#include<iostream>
using namespace std;
#define int long long
const int N=3000+10;
int n,m,k,h;
int a[N],b[N],dp[N][N];
signed main()
{
	ios::sync_with_stdio(0);cin.tie(0);cout.tie(0);
	cin>>n>>h>>m;
	for(int i=1;i<=n;i++)
	cin>>a[i]>>b[i];
	for(int i=1;i<=n;i++)
	{
		for(int j=0;j<=h;j++)dp[i][j]=-1;
	}
	for(int i=0;i<=h;i++)dp[0][i]=m;
	for(int i=1;i<=n;i++)
	{
		for(int j=0;j+a[i]<=h;j++)dp[i][j+a[i]]=dp[i-1][j];
		for(int j=0;j<=h;j++)
		{
			if(dp[i-1][j]>=b[i])
				dp[i][j]=max(dp[i][j],dp[i-1][j]-b[i]);
		}
	}
	int ans=0;
	for(int i=0;i<=n;i++)
	{
		for(int j=0;j<=h;j++)
		{
			if(dp[i][j]>=0)ans=max(ans,i);
		}
	}
	cout<<ans<<endl;
	return 0;
}
