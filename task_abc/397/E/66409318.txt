#include <bits/stdc++.h>
using namespace std;

#define ll long long
#define pii pair<int,int>
#define pll pair<ll, ll>
#define fs first
#define sc second

const int mxn = 2e5+10;
int N, K;
vector<int> tree[mxn];
int dp[mxn];

void dfs(int now, int par) {
	dp[now] = 1;
	int cnt = 0;
	for(auto nxt:tree[now]) {
		if(nxt == par) continue;
		dfs(nxt, now);
		if(!dp[nxt]) continue;
		cnt++;
		dp[now] += dp[nxt];
	}
	if(cnt > 2 || dp[now] > K) {
		cout<<"No\n";
		exit(0);
	}
	if(dp[now] == K) {
		dp[now] = 0;
	}
	else {
		if(cnt >= 2) {
			cout<<"No\n";
			exit(0);
		}
	}
	return;
}

int main() {
	ios::sync_with_stdio(0);cin.tie(0);cout.tie(0);
	cin>>N>>K;
	for(int i = 0;i<N*K-1;i++) {
		int a,b;
		cin>>a>>b;
		tree[a].push_back(b);
		tree[b].push_back(a);
	}
	dfs(1,1);
	cout<<"Yes\n";
	return 0;
}
