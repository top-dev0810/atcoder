#include <bits/stdc++.h>
using namespace std;

int main(){
  int n;
  cin >> n;
  
  vector<vector<char>> s(n, vector<char>(n)); 
  
  for(int i=1;i<=n;i++){
    int j = n+1-i;
    if (i <= j){
      for(int k=i-1;k<j;k++){
        for(int l=i-1;l<j;l++){
          if (i%2==0) s.at(k).at(l) = '.';
          else s.at(k).at(l) = '#';
        }
      }
    }
  }
  for(int i=0;i<n;i++){
    for(int j=0;j<n;j++){
      cout << s.at(i).at(j);
    }
    cout << endl;
  }
}
