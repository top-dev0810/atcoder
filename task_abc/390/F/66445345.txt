using i128 = __int128;

const int N=3e5+10;
int b[N],L[N],R[N];
void solve()
{
    int n;
    cin>>n;
    vector<int>a(n+1);
    for(int i=1;i<=n;i++)
    {
    	cin>>a[i];
    }
    for(int i=1;i<=n;i++)
    {
    	L[i]=max(b[a[i]],b[a[i]-1]);
    	b[a[i]]=i;
    }
    for(int i=0;i<=n+1;i++)
    {
    	b[i]=n+1;
    }
    for(int i=n;i;i--)
    {
    	R[i]=b[a[i]-1];
    	b[a[i]]=i;
    }
    i64 ans=0;
    for(int i=1;i<=n;i++)
    {
    	ans+=(i64)(R[i]-i)*(i-L[i]);

    }
    cout<<ans<<"\n";
}
int main()
{
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int T=1;
    //cin >> T;
    while (T--)
    {
        solve();
    }
    return 0;
}
