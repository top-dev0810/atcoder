#include<bits/stdc++.h>
using namespace std;

inline void solve()
{
    int a[10],b[10];
    int n=5;
    for(int i=1;i<=n;++i)cin>>a[i],b[i]=a[i];
    sort(b+1,b+6);   
    vector<int> tmp;
    for(int i=1;i<=n;++i)
    {
        if(a[i]!=b[i])tmp.push_back(i);
    }
    if(tmp.size()!=2)cout<<"No";
    else 
    {
        if(abs(tmp[0]-tmp[1])==1)cout<<"Yes";
        else cout<<"No";
    }
}
signed main()
{
    ios::sync_with_stdio(0);
    return solve(),0;
}
