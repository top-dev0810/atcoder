#include<bits/stdc++.h>
using namespace std;
#define int long long
int n,a[100],b[100];
unordered_map<int,int> vis;
int ans;
inline void dfs(int x,int cnt)
{
    if(x==n+1)
    {
        int tmp=0;
        for(int i=1;i<=cnt;++i)tmp^=b[i];
        if(!vis[tmp])ans++;
        vis[tmp]=1;
        return ;
    }
    for(int i=1;i<=cnt;++i)
    {
        b[i]+=a[x];
        dfs(x+1,cnt);
        b[i]-=a[x];
    }
    b[cnt+1]+=a[x];
    dfs(x+1,cnt+1);
    b[cnt+1]-=a[x];
}
inline void solve()
{
    cin>>n;
    for(int i=1;i<=n;++i)cin>>a[i];
    dfs(1,0);
    cout<<ans;
}
signed main()
{
    ios::sync_with_stdio(0);
    return solve(),0;
}
