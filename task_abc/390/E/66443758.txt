#include <bits/stdc++.h>
using namespace std;
using i64 = long long;
using i128 = __int128;
#define sz(x) (int)x.size()
#define all(v) v.begin(), v.end()
#define ps(y) cout << fixed << setprecision(y)
/*
 /\_/\              
( o.o )     /\_/\   
 > ^ <     ( o.o )    
            > ^ <   

 /\_/\      /\_/\  
( o.o )    ( o.o )  
 > ^ <      > ^ <       
*/
const int N=5010;
int n,m,f[3][N];
vector<pair<int,int>>seq[3];
void solve()
{
    cin>>n>>m;
    for(int i=0;i<n;i++)
    {
    	int t,x,y;
    	cin>>t>>x>>y;
    	t--;
    	seq[t].push_back({x,y});
    }
    memset(f,-63,sizeof(f));
    for(int t=0;t<3;t++)
    {
    	f[t][0]=0;
    	for(auto [x,y]:seq[t])
    	{
    		for(int i=m;i>=y;i--)
    		{
    			f[t][i]=max(f[t][i],f[t][i-y]+x);
    		}
    	}
    	for(int i=1;i<=m;i++)
    	{
    		f[t][i]=max(f[t][i],f[t][i-1]);
    	}
    }
    int ans=0;
    for(int i=0;i<=m;i++)
    {
    	for(int j=0;j+i<=m;j++)
    	{
    		ans=max(ans,min({f[0][i],f[1][j],f[2][m-i-j]}));
    	}
    }
    cout<<ans<<"\n";
}	
int main()
{
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int T=1;
   // cin >> T;
    while (T--)
    {
        solve();
    }
    return 0;
}
