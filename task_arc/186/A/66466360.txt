	template<typename T> inline bool read(T &s) { s = 0; int w = 1; char ch; while (ch = getchar(), !isdigit(ch) and (ch != EOF)) if (ch == '-') w = -1; if (ch == EOF) return false; while (isdigit(ch)) s = s * 10 + ch - 48, ch = getchar(); return s *= w, true; }

const int maxn=30;
const int maxm=900;
int n,q,k;
bitset <maxm+1> dp[maxn+5][maxn+5],Ans;
signed main(){
	cin>>n>>q;
	dp[0][0][0]=1;
	for(int L=0;L<=n;L++){
		for(int R=0;R<=n;R++){
			for(int u=2;L+u<=n;u++){
				for(int v=2;R+v<=n;v++){
					dp[L+u][R+v]|=(dp[L][R]<<(u*v));
				}
			}
			Ans|=dp[L][R];
		}
	}
	while(q--){
		cin>>k;
		if(Ans[n*n-k])cout<<"Yes\n";
		else cout<<"No\n";
	}
	return 0;
}
