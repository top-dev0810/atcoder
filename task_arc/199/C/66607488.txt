const int N=505,MOD=998244353;
int n,m,a[N][N];
int dp1[N][N],dp2[N][N];
int mx[N],mn[N];

int main()
{
	ios::sync_with_stdio(false),cin.tie(nullptr);
//	int _;cin>>_;while(_--)

	cin>>n>>m;
	for(int i=1;i<=m;i++)
		for(int j=1;j<=n;j++)cin>>a[i][j];
	for(int i=1;i<=n;i++)a[0][a[1][i]]=i;
	for(int i=1;i<=m;i++)
	{
		for(int j=1;j<=n;j++)a[n+1][j]=a[i][a[0][j]];
		for(int j=1;j<=n;j++)a[i][j]=a[n+1][j];
	}
	for(int i=2;i<=m;i++)
		for(int j=n;j;j--)a[i][j]=(a[i][j]-a[i][1]+n)%n+1;
	for(int i=1;i<=m;i++)dbg(VI(a[i]+1,a[i]+n+1));
	for(int i=0;i<=n;i++)dp1[i+1][i]=dp2[i+1][i]=1;
	for(int i=n;i>1;i--)
	{
		memset(mx,0,sizeof(mx));
		memset(mn,0x3f,sizeof(mn));
		for(int j=i;j<=n;j++)
		{
			bool ok=1;
			for(int k=1;k<=m;k++)
				ok&=(mx[k]=max(mx[k],a[k][j]))-(mn[k]=min(mn[k],a[k][j]))==j-i;
			for(int k=i;k<=j;k++)
			{
				if(ok)dp1[i][j]=(dp1[i][j]+1ll*dp2[i][k-1]*dp2[k+1][j])%MOD;
				dp2[i][j]=(dp2[i][j]+1ll*dp1[i][k]*dp2[k+1][j])%MOD;
			}
		}
	}
	cout<<dp2[2][n]<<endl;

	return 0;
}
