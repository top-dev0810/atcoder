#include <bits/stdc++.h>
using namespace std;
int main(){
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    int t;
    cin >> t;
    while(t--){
        int n,a,ma = 0;
        cin >> n;
        bool h = 0;
        vector<int> v(1),s(1);
        for(int i = 1; i <= n; i++){
            cin >> a;
            if(a == i && ma < a)
                h = 1;
            ma = max(ma,a);
            v.push_back(a);
            s.push_back(a);
        }

        if(h){
            sort(s.begin(),s.end());
            if(v == s)
                cout << 0 << '\n';
            else
                cout << 1 << '\n';
        }
        else{
            if(v.back() == 1 && v[1] == n)
                cout << 3 << '\n';
            else
                cout << 2 << '\n';
        }
    }
    return 0;
}
