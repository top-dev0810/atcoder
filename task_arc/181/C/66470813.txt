#include<bits/stdc++.h>
#define Maxn 505
using namespace std;
int n,a[Maxn],b[Maxn],ans[Maxn][Maxn];
int main(){
	scanf("%d",&n);
	for(int i=1; i<=n; i++) scanf("%d",a+i);
	for(int i=1; i<=n; i++) scanf("%d",b+i);
	for(int i=1; i<=n; i++){
		for(int j=0; j<i; j++){
			ans[a[i]][b[n-j]] ^= 1;
		}
	}
	
	for(int i=1; i<=n; i++){
		for(int j=1; j<=n; j++) printf("%d",ans[i][j]);
		printf("\n");
	}
	return 0;
}
