#include<bits/stdc++.h>
using namespace std;
int n,a[100100],b[100100];
int f[1010][1010];
int main(){
    cin>>n;
    for (int i=1;i<=n;i++)
    cin>>a[i];
    for (int i=1;i<=n;i++)
    cin>>b[i];
    reverse(b+1,b+n+1);
    for (int i=1;i<=n;i++){
        for (int j=1;j<=i;j++)
        f[a[i]][b[j]]=1;
    }
    for (int i=1;i<=n;i++,cout<<"\n")
    for (int j=1;j<=n;j++)
    cout<<f[i][j];
}
