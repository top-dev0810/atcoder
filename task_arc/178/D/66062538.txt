#include<bits/stdc++.h>
#define int long long
using namespace std;
const int P=998244353;
int a[510],p[510],dp[510][510][510];
void add(int &a,int b) {
	a=(a+b)%P;
}
signed main() {
	int n,m;
	cin>>n>>m;
	for(int i=1; i<=m; i++)cin>>a[i],p[a[i]]=i;
	if(!p[0])for(int i=1; i<=m+1; i++)dp[i][i][0]=1;
	else dp[p[0]][p[0]+1][0]=1;
	for(int i=1; i<n; i++) {
		if(!p[i]) {
			for(int r=1; r<=m+1; r++) {
				int s=0;
				for(int l=r; l; l--)add(s,dp[l][r][i-1]),add(dp[l][r][i],s);
			}
			for(int l=1; l<=m+1; l++) {
				int s=0;
				for(int r=l; r<=m+1; r++)add(s,dp[l][r][i-1]),add(dp[l][r][i],s);
			}
		} else {
			for(int l=1; l<=m+1; l++)for(int r=l; r<=m+1; r++)add(dp[min(p[i],l)][max(p[i]+1,r)][i],dp[l][r][i-1]);
		}
	}
	cout<<dp[1][m+1][n-1];
	return 0;
}
