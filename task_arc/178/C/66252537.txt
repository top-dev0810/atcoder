#include<bits/stdc++.h>
using namespace std;
const int N=200005;
int q,l,dp[N];
int main(){
    scanf("%d%d",&q,&l);
    for(int i=1;i<N;i++){
        dp[i]=N;
        for(int j=1;i>=j*(l-j)&&j<=l/2;j++){
            dp[i]=min(dp[i],dp[i-j*(l-j)]+1);
        }
    }
    for(int x;q;q--){
        scanf("%d",&x);
        if(dp[x]==N)printf("-1\n");
        else printf("%d\n",dp[x]);
    }
}
