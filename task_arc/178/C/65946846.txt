#include <bits/stdc++.h>
using namespace std;

const int maxn = 2e5+100;
int n,l,a[maxn+100],dp[maxn+100],xs[maxn+100],mx = INT_MIN; 
int h[maxn+100];

int main(){
	scanf("%d %d",&n,&l);
	int pre = -l+1;
	for(int i = 1;i <= n;i++) xs[i] = pre,pre += 2;
	h[n] = xs[n]; 
	for(int i = n-1;i >= 1;i--) h[i] = h[i+1] + xs[i];
	for(int i = 1;i <= maxn;i++){ 
		dp[i] = maxn;
		for(int j = 1;j*(l-j) <= i && j <= l/2;j++)
			dp[i] = min(dp[i],dp[i-j*(l-j)]+1);
	}
	for(int i = 1;i <= n;i++){
		scanf("%d",&a[i]);
		if(dp[a[i]] == maxn) printf("-1\n");
		else printf("%d\n",dp[a[i]]);
	}
	return 0;
}
