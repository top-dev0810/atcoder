#include<bits/stdc++.h>
using namespace std;
const int N=5e5+5,mod=998244353;
#define int long long
int h,w,ans;
int p[N],np[N];
int fst(int a,int b) {
	int rs=1;
	while(b) {
		if(b&1) rs=rs*a%mod;
		a=a*a%mod;
		b>>=1;
	}
	return rs;
}
void init() {
	p[0]=1;
	for(int i=1;i<=N-5;i++) p[i]=p[i-1]*i%mod;
	np[N-5]=fst(p[N-5],mod-2);
	for(int i=N-5;i>=1;i--) np[i-1]=np[i]*i%mod;
}
int C(int n,int m) {
	return p[n]*np[m]%mod*np[n-m]%mod;
}
signed main() {
	init();
	cin>>h>>w;
	if(h%2==0) {
		cout<<0;
		return 0;
	}
	if(w&1) {
		for(int i=0;i<=h;i++) {
			int j=abs(2*i-h);
			if(!j) j=w;
			if(__gcd(j,w)==1) ans=(ans+C(h,i))%mod;
		}
	} else {
		w/=2;
		for(int i=-h;i<=h;i++) {
			int j=abs(i)+w;
			if(__gcd(j,w)==1) {
				int az=abs(i);
				ans=(ans+C(2*h,h+az))%mod;
			}
		}
	}
	cout<<ans;
	return 0;
}
