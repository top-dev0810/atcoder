#include<bits/stdc++.h>
using namespace std;
const int N=105,M=1001,mod=1e9+7;
#define ll long long
ll dp[N][2005],a[N],b[N],g[2005];;
int n;
int main(){
    scanf("%d",&n);
    for(int i=1;i<=n;i++){
        scanf("%lld",&a[i]);
        b[i]=b[i-1]+a[i];
    }
    dp[0][M]=1;
    for(int i=1;i<=n;i++){
        for(int j=0;j<=(M<<1);j++){
            dp[i][j]+=dp[i-1][j];
            if(j!=M){
                dp[i][j+a[i]]=(dp[i][j+a[i]]+dp[i-1][j]+g[j])%mod;
            }
        }
        g[a[i]+M]=dp[i-1][M]%=mod;
    }
    ll ans=0;
    for(int i=0;i<=(M<<1);i++)ans=(ans+dp[n][i])%mod;
    cout<<ans;
}
