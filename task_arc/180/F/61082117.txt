#include <bits/stdc++.h>
#define Mod 1000000007
using namespace std;
void add(int &a,int b){a+=b;if(a>=Mod) a-=Mod;}
int qpow(int a,int p)
{
	int ret=1;
	for(;p;p>>=1,a=1ll*a*a%Mod)
		if(p&1) ret=1ll*ret*a%Mod;
	return ret;
}
int fac,n,A;
int b[10010],inv[10010];
int main()
{
	cin>>n>>A;

	for(int i=0;i<=n;i++)
		inv[i]=qpow(1ll*i*(A+1)%Mod+1,Mod-2);

	b[0]=1;
	for(int i=1;i<=n;i++)
	{
		for(int j=0;j<i;j++)
			add(b[i],1ll*b[j]*b[i-j-1]%Mod*(j+1)%Mod*inv[j]%Mod);
		b[i]=1ll*b[i]*qpow(i,Mod-2)%Mod;
	}
	fac=1;
	for(int i=1;i<=n;i++)
		fac=1ll*fac*i%Mod;
	cout<<1ll*b[n]*fac%Mod<<endl;
	return 0;
}
