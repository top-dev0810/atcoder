#include <bits/stdc++.h>
using namespace std;

int n,k,p[1000],t[1000],jl[1000];
vector<int> vec,ans1,ans2;

int main(){
	ios::sync_with_stdio(0);cin.tie(0);
	cin >> n >> k;
	for(int i = 1;i <= n;i++){
		cin >> p[i];
		t[p[i]] = i;
	}
	bool flag = 1;
	while(flag != 0){
		flag = 0;
		for(int i = 1;i <= n;i++){
			int pret = t[i];
			for(int j = i-1;j >= 1;j--){
				if(t[j] - t[i] >= k){
					ans1.push_back(min(t[i],t[j]));
					ans2.push_back(max(t[j],t[i]));
					swap(p[t[i]],p[t[j]]);
					swap(t[i],t[j]);
					flag = 1;
				}	
			}
			if(flag == 1) break;
		}
	}
	
	cout << ans1.size() << '\n';
	for(int i = 0;i < ans1.size();i++){
		cout << ans1[i] << " " << ans2[i] << '\n';
	}
	return 0;
}
