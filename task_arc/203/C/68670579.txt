#include<bits/stdc++.h>
#define int long long
using namespace std;
const int N=4e5+10,mod=998244353;
int T,n,m,k;
int jc[N],ny[N];
int qm(int a,int b){
	int sum=1;
	while(b){
		if(b&1)sum=sum*a%mod;
		a=a*a%mod;b>>=1;
	}return sum;
}
int C(int x,int y){return jc[x]*ny[y]%mod*ny[x-y]%mod;}
signed main(){
	jc[0]=ny[0]=1;
	for(int i=1;i<N;i++){
		jc[i]=jc[i-1]*i%mod;
		ny[i]=qm(jc[i],mod-2);
	}cin>>T;while(T--){
		cin>>n>>m>>k;
		int x=2*(n-1)*(m-1);
		if(k<n+m-2)cout<<0<<'\n';
		else if(k==n+m-2)cout<<C(n+m-2,n-1)<<'\n';
		else if(k==n+m-1)cout<<C(n+m-2,n-1)*(x%mod)%mod<<'\n';
		else{int ans=0;
			ans+=C(n+m-2,n-1)*((x/2%mod)*((x-1)%mod)%mod)%mod;
			ans-=C(n+m-4,n-2)*(n+m-3)%mod;
			if(m>=3)ans+=(n-1)*C(n+m-2,m-3)%mod;
			if(n>=3)ans+=(m-1)*C(n+m-2,n-3)%mod;
			cout<<(ans%mod+mod)%mod<<'\n';
		}
	}
	return 0;
}
