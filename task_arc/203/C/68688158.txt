#include <bits/stdc++.h>

#define int long long

using namespace std;

constexpr int mod = 998244353;

int t, n, m, k, jc[800005];

int ksm(int a, int b)
{
    int cnt = 1;
    while (b)
    {
        if (b & 1)
            cnt = cnt * a % mod;
        b >>= 1, a = a * a % mod;
    }
    return cnt;
}
int C(int x, int y)
{
    if (y < x)
        return 0;
    return jc[y] * ksm(jc[x] * jc[y - x] % mod, mod - 2) % mod;
}
signed main()
{

    jc[0] = 1;
    for (int i = 1; i <= 800000; i++)
        jc[i] = jc[i - 1] * i % mod;
    cin >> t;
    while (t--)
    {
        cin >> n >> m >> k;
        int ji = (n * (m - 1) % mod + m * (n - 1) % mod - (n + m - 2) + mod) % mod;
        if (k < n + m - 2)
            cout << 0ll << endl;
        else if (k == n + m - 2)
            cout << C(n - 1, m + n - 2) << endl;
        else if (k == n + m - 1)
            cout << C(n - 1, m + n - 2) * ji % mod << endl;
        else
        {
            int ans = C(n - 1, n + m - 2) * ji % mod * (ji - 1) % mod * ((mod + 1) / 2) % mod;
            ans -= C(n - 2, n + m - 4) * (n + m - 3) % mod, ans = (ans + mod) % mod;
            ans += (n - 1) * (C(n, n + m - 3) + C(n + 1, n + m - 3)) % mod + (m - 1) * (C(m, n + m - 3) + C(m + 1, n + m - 3)) % mod, ans %= mod;
            cout << ans << endl;
        }
    }
}
