#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
#define int ll
#define ff first
#define ss second
#define pb push_back
#define eb emplace_back
typedef pair<int, int> pii;
const ll inf=3e18;
const int maxn=1e6+10;

void fill_h(int a, int b, int l, int&n, int&m, vector<pii>&ansh, vector<pii>&ansv)
{
    n--;
    ansh.eb(a, b);
}

void fill_v(int a, int b, int l, int&n, int&m, vector<pii>&ansh, vector<pii>&ansv)
{
    m--;
    ansv.eb(a, b);
}

void fill_lxl(int a, int b, int l, int&n, int&m, vector<pii>&ansh, vector<pii>&ansv)
{
    if(n >= l)
    {
        for(int i=a; i<=a+l-1; i++)
        {
            n--;
            ansh.eb(i, b);
        }
        return;
    }
    assert(m >= l);
    for(int j=b; j<=b+l-1; j++)
    {
        m--;
        ansv.eb(a, j);
    }
}

void fill_multiple_l(int a, int b, int c, int d, int l, int&n, int&m, vector<pii>&ansh, vector<pii>&ansv)
{
    for(int i=a; i<=c; i+=l)
    {
        for(int j=b; j<=d; j+=l)
        {
            fill_lxl(i, j, l, n, m, ansh, ansv);
        }
    }
}
