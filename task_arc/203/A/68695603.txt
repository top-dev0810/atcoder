#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
typedef pair<int,int> pii;
int n , m , k;
void solv()
{
    cin >> n >> m;
    if(m % 2 == 0) {
        cout << 1LL * n * (m / 2) << '\n';
    }
    else {
        cout << 1LL * n * (m / 2) + 1 << '\n';
    }
    return;
}
int main()
{
    ios::sync_with_stdio(false) ; cin.tie(0) ; cout.tie(0) ;
    int T;cin >> T;
    while(T--) solv();
    return 0;
}
