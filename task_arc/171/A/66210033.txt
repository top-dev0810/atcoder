const int inf=1e18;
const int mod=998244353;
int fast(int x,int k,int mod)
{
	int ret=1;
	while(k)
	{
		if(k&1) ret=ret*x%mod;
		x=x*x%mod;
		k>>=1;
	}
	return ret;
}
void solve()
{
	int n,a,b;
	cin>>n>>a>>b;
	if(a>n)
	{
		cout<<"No\n";
		return;
	}
	if(a<=n/2)
	{
		if(b<=((n+1)/2)*(n-a)) cout<<"Yes\n";
		else cout<<"No\n";
	}
	else
	{
		if(b<=(n-a)*(n-a)) cout<<"Yes\n";
		else cout<<"No\n";
	}
}
signed main()
{
	ios::sync_with_stdio(0);
	cin.tie(0);
	int T=1;
	cin>>T;
	while(T--) solve();
	return 0;
}
