int qpow(int n,int k)
{
    int res=1;
    while(k)
    {
        if(k&1)(res*=n)%=MOD;
        (n*=n)%=MOD;
        k>>=1;
    }
    return res;
}

int p,base,n,m;

int b[1000010],flg[1000010],f[1000010];

void work()
{
    cin>>p>>base>>n>>m;
    for(int i=1;i<=m;i++)
    {
        int l,r;
        cin>>l>>r;
        b[l-1]|=(1<<r);
        b[r]|=(1<<l-1);
    }
    n++;
    for(int i=0;i<1<<n;i++)
    {
        flg[i]=1;
        for(int j=0;j<n;j++)
        {
            if(((i>>j)&1)&&i&b[j])
            {
                flg[i]=0;
                break;
            }
        }
    }
    memset(f,0x3f,sizeof f);
    f[0]=0;
    for(int i=0;i<(1<<n);i++)
    {
        for(int t=i;t;t=(t-1)&i)
        {
            if(flg[t])f[i]=min(f[i],f[t^i]+1);
        }
    }
    if(f[(1<<n)-1]<=p)
    {
        cout<<"Yes\n";
    }
    else
    {
        cout<<"No\n";
    }
}

signed main()
{
#ifndef cin

#endif
#endif
    int t=1;
    // cin>>t;
    while(t--)
    {
        work();
    }
}
