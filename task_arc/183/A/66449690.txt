#include <bits/stdc++.h>
using namespace std;

int main(){
  int n,m; cin >> n >> m;
  if (n%2==0){
    vector<int> A(n,m); --A[n/2-1]; cout << n/2 << ' ';
    for (int i(n);i;--i) for (int k(0);k < A[i-1];++k) cout << i << ' ';
    cout << endl;
  } else {
    vector<int> A(n,m); A[n/2] = 0; for (int i(0);i < m;++i) cout << n/2+1 << ' ';
    if (n>1){ --A[n/2-1]; cout << n/2 << ' '; }
    for (int i(n);i;--i) for (int k(0);k < A[i-1];++k) cout << i << ' ';
    cout << endl;
  }
}
