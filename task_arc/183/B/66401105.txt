int qpow(int n,int k)
{
    int res=1;
    while(k)
    {
        if(k&1)(res*=n)%=MOD;
        (n*=n)%=MOD;
        k>>=1;
    }
    return res;
}

vector<int>gtfrm(vector<int>res)
{
    vector<int>ans;
    int lst=-1;
    for(auto x:res)
    {
        if(x==lst)continue;
        ans.push_back(x);
        lst=x;
    }
    return ans;
}

bool ck(vector<int>a,vector<int>b)
{
    int n=a.size(),m=b.size();
    if(m>n)return false;
    int j=0;
    for(int i=0;i<n;i++)
    {
        if(a[i]==b[j])
        {
            j++;
            if(j==m)return true;
        }
    }
    return false;
}

void work()
{
    int n,k;
    cin>>n>>k;
    vector<int>a(n),b(n);
    read(a);
    read(b);
    if(a==b){cout<<"Yes\n";return ;}
    if(k==1)
    {
        if(ck(gtfrm(a),gtfrm(b)))
        {
            cout<<"Yes\n";
        }
        else
        {
            cout<<"No\n";
        }
        return ;
    }
    set<int>se;
    for(auto x:a)se.insert(x);
    for(auto x:b)
    {
        if(!se.count(x))
        {
            cout<<"No\n";
            return ;
        }
    }
    map<int,int>lst;
    for(int i=0;i<n;i++)
    {
        if(!lst.count(b[i]))
        {
            lst[b[i]]=i;
        }
        else
        {
            int dis=i-lst[b[i]];
            if(dis<=k)
            {
                cout<<"Yes\n";
                return ;
            }
            lst[b[i]]=i;
        }
    }
    cout<<"No\n";
}
