int H,W,C;
int f[maxn][maxn];
int g[maxn][maxn];
int p[maxn*2];
int c[maxn][maxn];
const ll v[2] = {mod-1,1};
void Solve() {
	cin >> H >> W >> C;
	for(int i = 0; i <= max(H,W); i++) {
		for(int j = 0; j <= i; j++) {
			if(!j) c[i][j] = 1;
			else c[i][j] = (c[i-1][j-1] + c[i-1][j]) % mod;
		}
	}
	p[0] = 1; for(int i = 1; i <= H+W+1; i++) p[i] = 1ll * p[i-1] * C % mod;
	ll ans = 0;
	for(int i=0;i<=H;i++) {
		for(int j=0;j<=W;j++) {
			if(i == 0 || j == 0) {
				f[i][j] = 1; g[i][j] = 1;
				continue;
			}
			for(int k = 1; k <= i; k++) {
				f[i][j] = (f[i][j] + v[k&1] * f[i-k][j] % mod * c[i][k] % mod * p[k] % mod) % mod;
			}
			for(int k = 1; k <= j; k++) {
				f[i][j] = (f[i][j] + v[k&1] * f[i][j-k] % mod * c[j][k] % mod * p[k] % mod) % mod;
			}
			for(int k = 1; k <= i; k++) {
				f[i][j] = (f[i][j] + v[(k+1)&1] * g[i-k][j] % mod * c[i][k] % mod * C % mod) % mod;
			}
			for(int k = 1; k <= j; k++) {
				g[i][j] = (g[i][j] + v[k&1] * f[i][j-k] % mod * c[j][k] % mod) % mod;
			}
		}
	}
	for(int i=1;i<=H;i++) {
		for(int j=1;j<=W;j++) {
			ans = (ans + v[(i+j+1)&1] * f[H-i][W-j] % mod * c[H][i] % mod * c[W][j] % mod * C % mod) % mod;
//			cerr<<f[i][j]<<' ';
//			cerr<<g[i][j]<<' ';
		}
//		cerr<<'\n';
	}
	cout<<ans<<'\n';
}
