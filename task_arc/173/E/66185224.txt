const int V=61;

struct BL {
	int b[100];
	int cnt=0;
	void insert(int x) {
		
		for(int i=V-1;i+1;i--) {
			if((x>>i)&1ll) {
				if(!b[i]) return b[i]=x,cnt++,void();
				x^=b[i];if(!x) return ;
			}
		}
	}
	
	int max() {
		int x=0;
		for(int i=V-1;i+1;i--) {
			x=std::max(x,x^b[i]);
		}
		return x;
	}
	
	void rebuild() {
		for(int i=0;i<V;i++) {
			if(!b[i]) continue;
//			cout<<i<<" "<<b[i]<<"\n";
			
			for(int j=i-1;j>=0;j--) {
				if((b[i]>>j&1ll)&&b[j]) b[i]^=b[j];
			}
			
//			cout<<i<<" "<<b[i]<<"\n";
		}
	}
	
	
	
}B;

int n;
int a[M];

void solve() {
	read(n);
	
	int sum=0;
	
	for(int i=1;i<=n;i++) read(a[i]),sum^=a[i];
	
//	cout<<sum<<"\n";
	
	for(int i=1;i<=n;i++) {
		for(int j=i+1;j<=n;j++) {
			B.insert(a[i]^a[j]);
		}
	}
	
//	cout<<B.cnt<<"\n";
	
	if((n&1)||n<=2) cout<<B.max();
	
	else {
		
//		cout<<sum<<"\n";
		
		if(sum!=B.max()||(sum==B.max()&&B.cnt!=n-1)||(sum==B.max()&&B.cnt==n-1&&n%4==0)) cout<<B.max();
		else {
			B.rebuild();
			int c=B.cnt;
			int ans=0;
			for(int i=V-1;i+1;i--) {
				if(c!=1&&B.b[i]) ans^=B.b[i],c--;
			}
			cout<<ans;
		}
	}
	
}
