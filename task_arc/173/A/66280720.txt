const int mod=998244353;
int fast(int x,int k,int mod)
{
	int ret=1;
	while(k)
	{
		if(k&1) ret=ret*x%mod;
		x=x*x%mod;
		k>>=1;
	}
	return ret;
}
void solve()
{
	int n;cin>>n;
	int len=1,sum=0;
	int base=1;
	for(int i=1,pw=9;i<=20;++i)
	{
		base=base*pw;
		if(sum+base>=n)
		{
			len=i;break;
		}
		sum+=base;
	}
	int pre=-1;
	for(int i=1;i<=len;++i)
	{
		base=base/9;
		for(int c=0;c<=9;++c)
		{
			if(i==1&&c==0) continue;
			if(c==pre) continue;
			if(sum+base>=n)
			{
				pre=c;
				cout<<c;
				break;
			}
			sum+=base;
		}
	}
	cout<<'\n';
}
signed main()
{
	ios::sync_with_stdio(0);
	cin.tie(0);
	int T=1;
	cin>>T;
	while(T--) solve();
	return 0;
}
