#include<bits/stdc++.h>

#define ll long long
#define pb push_back
int t,m,k;
int main(){
	scanf("%d",&t);
	while(t--){
		scanf("%d%d",&m,&k);
		if(k==m){
			printf("Yes\n%d\n",m);
			for(int i=0;i<m;i++) printf("%d ",i);printf("\n"); 
		}
		else if(k%2==1){
			printf("Yes\n%d\n",k/2+1);
			for(int i=0;i<=k/2;i++) printf("%d ",i);printf("\n");
		}
		else if(k%2==0&&k>=6){
			printf("Yes\n%d\n",k/2);
			for(int i=0;i<=k/2;i++) if(i!=1) printf("%d ",i);
			printf("\n");
		}
		else if(k==2&&m%2==0){
			printf("Yes\n2\n%d %d\n",0,m/2);
		}
		else if(k==4&&m%4==0){
			printf("Yes\n4\n%d %d %d %d\n",0,m/4,m/2,m/2+m/4);
		}
		else printf("No\n");
	}
	return 0;
}
