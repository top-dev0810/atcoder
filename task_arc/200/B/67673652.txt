#include<bits/stdc++.h>
using namespace std;
#define int long long
int a1,a2,a3;
inline int read(){
	register int s = 0,w = 1;
	char ch = getchar();
	while(!isdigit(ch)){
		if(ch == '-')w = -1;
		ch = getchar();
	}
	while(isdigit(ch)){
		s = (s << 3) + (s << 1) + (ch ^ 48);
		ch = getchar();
	}
	return s * w;
}
int ksm(int x,int y){
	int ans = 1;
	while(y){
		if(y & 1)ans *= x;
		y >>= 1;
		x *= x;
	}
	return ans;
}
void Readinit(){
	a1 = read();
	a2 = read();
	a3 = read(); 
}
void solve(){
	int flag = 1;
	if(a1 > a2){
		swap(a1,a2);
		flag = 0;
	}
	if(a2 > a3 || a3 > a1 + a2){
		printf("No\n");
		return;
	}
	printf("Yes\n");
	int x1,x2;
	if(a1 + a2 == a3){
		x1 = 2 * ksm(10,a1 - 1) + 1;
		x2 = 5 * ksm(10,a2 - 1);
	}else{
		x1 = ksm(10,a1 - 1);
		x2 = ksm(10,a2) - ksm(10,a1 + a2 - a3 - 1);
	}
	if(flag == 0){
		swap(x1,x2);
	}
	printf("%lld %lld\n", x1,x2);
}
signed main(){
	int T = read();
	while(T--){
		Readinit();
		solve();
	}
	return 0;
}
