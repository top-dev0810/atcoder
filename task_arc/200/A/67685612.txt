void solve(){

	int pos1 = 0,pos2 = 0;
	for(int i = 1;i <= n;i++){
		cin >> b[i];
		s[i] = a[i]*1.0/b[i];
		if(s[i] < mi) pos1 = i;
		mi = min(mi,s[i]);
	}
	bool f = 1;

	for(int i = 1;i <= n;i++) if(s[i] > mi) f = 0,pos2 = i;
	if(f){
		cout << "No" << endl;
		return;
	}
	cout << "Yes" << endl;

	for(int i = 1;i <= n;i++){
		if(i==pos1){
			cout << -a[pos2]-b[pos2] << " ";
		}else if(i==pos2){
			cout << a[pos1] + b[pos1] << " ";
		}else{
			cout << 0 << " ";
		}
	}

}
