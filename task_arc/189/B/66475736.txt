bool Mst;
const int Max=2e5+10;

const int inf=1e9+10;

inline int read(){
   int res=0,v=1;
   char c=getchar();
   while(c<'0'||c>'9'){v=(c=='-'?-1:1);c=getchar();}
   while(c>='0'&&c<='9'){res=(res<<3)+(res<<1)+(c^48);c=getchar();}
   return res*v;
}


int a[Max];
vector<int>b,c;


bool Med;
signed main(){
    int n=read();
    for(int i=1;i<=n;++i)a[i]=read();
    for(int i=2;i<=n;++i){
        if(i&1)b.pb(a[i]-a[i-1]);
        else c.pb(a[i]-a[i-1]);
    }
    sort(b.begin(),b.end());sort(c.begin(),c.end());
    int cb=0,cc=0,ans=a[1];
    for(int i=2;i<=n;++i){
        int val=0;
        if(i&1){
            val=b[cb];
            ++cb;
        }else {
            val=c[cc];++cc;
        }
        a[i]=a[i-1]+val;
        ans+=a[i];
    }
    cout << ans << "\n";
    cerr<< "Time: "<<clock()/1000.0 << "s\n";
    cerr<< "Memory: " << (&Mst-&Med)/1000000.0 << "MB\n";
    return 0;
}
