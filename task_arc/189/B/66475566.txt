template<typename T>
void read(vector<T> &vec)
{
    for(auto &x:vec)
    {
        cin>>x;
    }
}

int qpow(int n,int k)
{
    int res=1;
    while(k)
    {
        if(k&1)(res*=n)%=MOD;
        (n*=n)%=MOD;
        k>>=1;
    }
    return res;
}

int n;

void work()
{
    cin>>n;
    vector<int>vec(n);
    read(vec);
    vector<int>cf1,cf2;
    for(int i=0;i+1<n;i+=2)
    {
        cf1.push_back(vec[i+1]-vec[i]);
    }
    for(int i=1;i+1<n;i+=2)
    {
        cf2.push_back(vec[i+1]-vec[i]);
    }
    sort(cf1.begin(),cf1.end());
    sort(cf2.begin(),cf2.end());
    vector<int>cf;
    for(int i=0;i<cf2.size();i++)
    {
        cf.push_back(cf1[i]);
        cf.push_back(cf2[i]);
    }
    if(cf1.size()>cf2.size())cf.push_back(cf1.back());
    int sum=vec[0],now=vec[0];
    for(auto x:cf)
    {
        now+=x;
        sum+=now;
    }
    cout<<sum<<endl;
}

#endif

    int t=1;
    // cin>>t;
    while(t--)
    {
        work();
    }
