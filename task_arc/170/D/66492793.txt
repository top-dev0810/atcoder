#include<bits/stdc++.h>
using namespace std;
const int N=2e5+5;
long long t,n,m,a[N],b[N],top,ans,sum,bj;
int main()
{
	ios::sync_with_stdio(false);
	cin.tie(0);
	cout.tie(0);
	cin>>t;
	while(t--)
	{
		cin>>n;
		for(int i=1;i<=n;++i) 
			cin>>a[i];
		for(int i=1;i<=n;++i)
			cin>>b[i];
		sort(a+1,a+n+1);
		sort(b+1,b+n+1);
		bj=0;
		a[n+1]=1e18;
		for(int i=1;i<=n;++i)
			if((a[i]-b[1]<a[i-1])||(a[i]+b[1])>a[i+1]) 
				bj=i;
		if(!bj)
			cout<<"Bob"<<'\n';
		else{
			top=n;
			for(int i=n;i>=1;--i)
			{
				while((top&&b[i]+a[bj]<=a[top])||top==bj)
					--top;
				if(b[i]-a[bj]>=a[top])
				{
					bj=0;
					cout<<"Bob"<<'\n';
					break;
				}
			}
			if(bj)
				cout<<"Alice"<<'\n';
		}
	}
	return 0;
}
