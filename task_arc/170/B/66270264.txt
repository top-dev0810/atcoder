int main()
{
 	ios_base::sync_with_stdio(0);
 	cin.tie(0);
 	cout.tie(0);
 	ll n;
 	cin >> n;
 	vector<ll> arr(n);
 	FOR(i,0,n)cin >> arr[i];
 	FOR(i,0,n)arr[i]--;
 	ll ans = 0;
    FOR(i,0,n)
	{
        ll pos = i;
        vector<ll> bad(10, 0);
        while (pos<n && !bad[arr[pos]])
		{
            FOR(j,i,pos)
			{
                ll c = 2 * arr[pos] - arr[j];
                if (c >= 0 && c < bad.size())
				{
                    bad[c] = 1;
                }
            }
            pos++;
        }
        ans += (n-pos);
    }
    cout << ans << endl;
 	return 0;
}
