#include<bits/stdc++.h>
using namespace std;
const int N=5e3+5,mod=998244353;
long long n,m,a[N],dp[N][N],ans;
int main()
{
	ios::sync_with_stdio(false);
	cin.tie(0);
	cout.tie(0);
	cin>>n>>m;
	ans=1;
	for(int i=1;i<=n;++i)
	{
		cin>>a[i];
		if(!a[i])
			ans=(ans*m)%mod;
	}
	if(m-1>=n)
	{
		cout<<ans;
		return 0;
	}
	dp[0][0]=1;
	for(int i=1;i<=n;++i)
	{
		for(int j=1;j<=m+1;++j)
		{
			if(a[i]==1)
				dp[i][j]=dp[i-1][j-1];
			else
				dp[i][j]=((dp[i-1][j-1]*(m-j+1))%mod+(dp[i-1][j]*j)%mod)%mod;
		}
	}
	ans=0;
	for(int i=1;i<=m+1;++i)
		ans+=dp[n][i],ans%=mod;
	cout<<ans;
	return 0;
}
