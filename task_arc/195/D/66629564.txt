#include<bits/stdc++.h>
using namespace std;
const int N=2005;
void solve(){
	int n;scanf("%d",&n);
	vector<int> a(n+2);
	for(int i=1;i<=n;++i)
		scanf("%d",&a[i]);
	vector<array<int, 2> > f(n+1);
	f[1][0]=1;
	f[1][1]=1e9;
	f[2][0]=(a[1]==a[2]?1:2);
	f[2][1]=f[2][0]+1;
	for(int i=3;i<=n;++i){
		f[i][0]=f[i][1]=1e9;
		f[i][0]=min(f[i][0],f[i-1][0]+(a[i-1]!=a[i]));
		f[i][0]=min(f[i][0],f[i-1][1]+(a[i-2]!=a[i]));
		f[i][1]=min(f[i][1],f[i-2][0]+(a[i-2]!=a[i])+(a[i]!=a[i-1])+1);
		f[i][1]=min(f[i][1],f[i-2][1]+(a[i-3]!=a[i])+(a[i]!=a[i-1])+1);
	}
	printf("%d\n",min(f[n][0],f[n][1]));
}
int main(){
	int T;cin>>T;
	while(T--)solve();
}
