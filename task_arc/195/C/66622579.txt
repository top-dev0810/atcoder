#include<bits/stdc++.h>
using namespace std;
int t;
int main(){
	cin>>t;
	while(t--){
		int r,b;
		cin>>r>>b;
		if(r%2==1){
			puts("No");
			continue;
		}
		if(r==0&&b%2==1){
			puts("No");
			continue;
		}
		puts("Yes");
		if(!r){
			int x=1,y=2;
			for(int i=1;i<=b/2;i++){
				cout<<"B "<<x<<' '<<y<<'\n';
				x++;y++;
			}
			y-=2;
			for(int i=1;i<=b/2;i++){
				cout<<"B "<<x<<' '<<y<<'\n';
				x--;
				y--;
			}
			continue;
		}
		int x=r/2,y=2;
		for(int i=1;i<=r/2;i++){
			cout<<"R "<<x<<' '<<y<<'\n';
			x--;
		}
		x=1;y=3;
		for(int i=1;i<=r/2;i++){
			cout<<"R "<<x<<' '<<y<<'\n';
			x++;
		}
		if(b==1){
			cout<<"B "<<r/2+1<<' '<<3<<'\n';
			continue;
		}
		if(b&1){
			x=r/2+1;y=3;
			for(int i=1;i<=b/2;i++){
				cout<<"B "<<x<<' '<<y<<'\n';
				x++;y++;
			}
			y-=2;
			for(int i=1;i<=b/2+1;i++){
				cout<<"B "<<x<<' '<<y<<'\n';
				x--;y--;
			}
		}
		else{
			x=r/2;y=4;
			for(int i=1;i<=b/2;i++){
				cout<<"B "<<x<<' '<<y<<'\n';
				x++;y++;
			}
			y-=2;
			for(int i=1;i<=b/2;i++)
			{
				cout<<"B "<<x<<' '<<y<<'\n';
				x--;
				y--;
			}
		}
	}
	return 0;
}
