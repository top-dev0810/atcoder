#include <bits/stdc++.h>
using namespace std;
int t;
int r,b;
void solve()
{
	cin>>r>>b;
	if(r%2==1)
	{
		cout<<"No"<<endl;
		return ;
	}
	if((r==0 && b%2==1))
	{
		cout<<"No"<<endl;
		return ;
	}	
	cout<<"Yes"<<endl;
	if(!r)
	{
		int x=1,y=2;
		for(int i=1;i<=b/2;i++)
		{
			cout<<"B"<<" "<<x<<" "<<y<<endl;
			x++;
			y++;
		}
		y-=2;
		for(int i=1;i<=b/2;i++)
		{
			cout<<"B"<<" "<<x<<" "<<y<<endl;
			x--;
			y--;
		}	
		return ;	
	}
	int x=r/2,y=2;
	for(int i=1;i<=r/2;i++)
	{
		cout<<"R"<<" "<<x<<" "<<y<<endl;
		x--;
	}
	x=1;
	y=3;
	for(int i=1;i<=r/2;i++)
	{
		cout<<"R"<<" "<<x<<" "<<y<<endl;
		x++;
	}
	if(b==1)
	{
		cout<<"B"<<" "<<r/2+1<<" "<<3<<endl;
		return ;
	}
	if(b&1)
	{
		x=r/2+1,y=3;
		for(int i=1;i<=b/2;i++)
		{
			cout<<"B"<<" "<<x<<" "<<y<<endl;
			x++;
			y++;
		}
		y-=2;
		for(int i=1;i<=b/2+1;i++)
		{
			cout<<"B"<<" "<<x<<" "<<y<<endl;
			x--;
			y--;
		}
	}
	else
	{
		int x=r/2,y=4;
		for(int i=1;i<=b/2;i++)
		{
			cout<<"B"<<" "<<x<<" "<<y<<endl;
			x++;
			y++;
		}
		y-=2;
		for(int i=1;i<=b/2;i++)
		{
			cout<<"B"<<" "<<x<<" "<<y<<endl;
			x--;
			y--;
		}
	}
}
int main()
{
	cin>>t;
	while(t--)
	{
		solve();
	}
 	return 0;
}
