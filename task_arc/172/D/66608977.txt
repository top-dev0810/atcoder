#include<bits/stdc++.h>
using namespace std;
int a[25][25];
int main(){
	int n;
	scanf("%d",&n);
	for(int i=1;i<=n;++i)a[i][i]=1e8;
	for(int i=1;i<=n*(n-1)/2;++i){
		int u,v;
		scanf("%d%d",&u,&v);
		a[u][v]=a[v][u]=n*(n-1)/2+1-i;
	}
	for(int i=1;i<=n;++i){
		for(int j=1;j<=n;++j)printf("%d ",a[i][j]);
		printf("\n");
	}
	return 0;
}
