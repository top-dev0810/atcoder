#include <bits/stdc++.h>
using namespace std;
const int MAXN = 25;
int a[MAXN][MAXN];
int main() {
	int n; scanf("%d", &n);
	for (int i = 1; i <= n; i++) a[i][i] = 100000000;
	for (int i = 1; i <= n * (n - 1) / 2; i++) {
		int a, b; scanf("%d %d", &a, &b); ::a[a][b] = n * (n - 1) / 2 - i;
	}
	for (int i = 1; i <= n; i++) for (int j = 1; j <= n; j++) printf("%d%c", a[i][j], " \n"[j == n]);
}
