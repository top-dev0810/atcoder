const int inf=1e18;
const int mod=998244353;
int fast(int x,int k)
{
	int ret=1;
	while(k)
	{
		if(k&1) ret=ret*x%mod;
		x=x*x%mod;
		k>>=1;
	}
	return ret;
}
void solve()
{
	int n,m,k;
	cin>>n>>k>>m;
	vector<int> fac(n+1),inv(n+1);
	fac[0]=inv[0]=1;
	for(int i=1;i<=n;++i) fac[i]=fac[i-1]*i%mod;
	inv[n]=fast(fac[n],mod-2);
	for(int i=n-1;i>=1;--i)
	{
		inv[i]=inv[i+1]*(i+1)%mod;
	}
	auto C=[&](int n,int m) -> int
	{
		if(n<m||m<0) return 0;
		return fac[n]*inv[m]%mod*inv[n-m]%mod;
	};
	int ans=1;
	int tmp=n-k+1;
	for(int i=1;i<=tmp;++i)
	{
		ans=ans*(m-i+1)%mod;
	}
	// cout<<ans<<"???"<<endl;
	ans=ans*fast(m-tmp+1,n-tmp)%mod;
	cout<<ans<<'\n';
}
signed main()
{
	ios::sync_with_stdio(0);
	cin.tie(0);
	int T=1;
	// cin>>T;
	while(T--) solve();
	return 0;
}
