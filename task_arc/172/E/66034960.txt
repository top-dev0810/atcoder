#include <bits/stdc++.h>
#define int long long
using namespace std;
const int mod=1e9;
const int Mod=1e6;
const int Max=1e6+10;
int ksm(int a,int b){
	int ans=1;
	for(;b;b>>=1){
		if(b&1)ans=ans*a%mod;
		a=a*a%mod;
	}
	return ans;
}

int ksm2(int a,int b){
	int ans=1;
	for(;b;b>>=1){
		if(b&1)ans=ans*a%Mod;
		a=a*a%Mod;
	}
	return ans;
}
int vis[Max];

signed main(){
	int q;
	scanf("%lld",&q);
	for(int i=1;i<=Mod;++i){
		vis[ksm2(i,i)]=i;
	}
	for(int i=1;i<=q;++i){
		int x;
		scanf("%lld",&x);
		int z=x%Mod;
		for(int j=0;j<=1e3;++j){
			int l=j*1e6+vis[z];
			if(ksm(l,l)==x){
				printf("%lld\n",l);
				break;
			}
		}
	}
	
	return 0;
}
