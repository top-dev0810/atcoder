const int inf=1e18;
const int mod=1e9;
int fast(int x,int k,int mod)
{
	int ret=1;
	while(k)
	{
		if(k&1) ret=ret*x%mod;
		x=x*x%mod;
		k>>=1;
	}
	return ret;
}
void solve()
{
	// int n;
	int x;
	cin>>x;
	vector<int> dp(20);
	for(int i=1;i<100;++i)
	{
		if(fast(i,i,100)==x%100)
		{
			dp[2]=i;
			break;
		}
	}
	int base=10;
	for(int i=3;i<=9;++i)
	{
		base*=10;
		for(int t=0;t<=9;++t)
		{
			dp[i]=t*base+dp[i-1];
			if(fast(dp[i],dp[i],base*10)==x%(base*10)) break;
		}
	}
	cout<<dp[9]<<'\n';
}
signed main()
{
	ios::sync_with_stdio(0);
	cin.tie(0);
	int T=1;
	cin>>T;
	while(T--) solve();
	return 0;
}
