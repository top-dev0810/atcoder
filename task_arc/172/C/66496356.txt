#include <bits/stdc++.h>
using namespace std;
#define endl '\n'
#define forr(i, a, b) for (int i = a; i <= b; ++i)
#define forl(i, a, b) for (int i = a; i >= b; --i)
#define lson p << 1
#define rson p << 1 | 1
typedef pair<int, int> PII;
using i64 = long long;
using u64 = unsigned long long;
using u32 = unsigned;
const int P = 911;
const int M = 998244353;
// 切勿水群焦躁
void solve()
{
    int n;
    cin >> n;
    string s;
    cin >> s;
    auto check = [&](int num) -> int
    {
        return (num == 0 ? 0 : (num > 0 ? 1 : 2));
    };
    vector<int> pre(n);
    forr(i, 1, n - 1)
    {
        pre[i] = pre[i - 1] + (s[i] == 'A' ? 1 : -1);
    }
    vector<u64> pp(n + 1), mi(n + 1), ps(n + 1);
    mi[0] = 1;
    pp[0] = 0;
    forr(i, 1, n)
    {
        mi[i] = mi[i - 1] * P;
    }
    forr(i, 1, n - 1)
    {
        pp[i] = pp[i - 1] * P + check(pre[i]);
    }
    forl(i, n - 1, 1)
    {
        ps[i] = ps[i + 1] + mi[n - i - 1] * check(pre[i] + (s[0] == 'A' ? 1 : -1));
    }
    unordered_set<u64> ust;
    forr(i, 0, n - 1)
    {
        ust.insert(pp[i] * mi[n - i] + check(pre[i] + (s[0] == 'A' ? 1 : -1)) * mi[n - i - 1] + ps[i + 1]);
    }
    cout << ust.size();
}
//=====================================
signed main()
{
#ifdef LOCAL

    clock_t c1 = clock();
#endif
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);
    //=====================================
    int t;
    t = 1;
    while (t--)
    {
        solve();
    }
//=====================================
#ifdef LOCAL
    cerr << "Time used: " << clock() - c1 << " ms" << endl;
#endif
    return 0;
}
