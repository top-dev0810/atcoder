#include<bits/stdc++.h>
#define N 60010
using namespace std;
int m,n,num[N][2],dp[2][21][21];
string s;
int main() {
	int i,j,k,x;
	char c;
	cin>>m>>n>>s;
	for(i=0; i<n; i++) {
		cin>>x>>c;
		num[x][c=='J']++;
	}
	memset(dp,63,sizeof(dp));
	dp[1][0][1]=dp[0][1][0]=0;
	for(i=0; i<=m; i++) {
		for(j=0; j<20; j++) {
			for(k=0; k<20; k++) {
				dp[0][j+1][k]=min(dp[0][j+1][k],dp[1][j][k]);
				dp[1][j][k+1]=min(dp[1][j][k+1],dp[0][j][k]);
			}
		}
		for(j=0; j<20; j++) {
			for(k=0; k<20; k++) {
				for(x=0; x<2; x++) {
					dp[x][max(0,j-num[i][0])][max(0,k-num[i][1])]=min(dp[x][max(0,j-num[i][0])][max(0,k-num[i][1])],dp[x][j][k]);
				}
			}
		}
		if(i==m) {
			break;
		}
		for(j=0; j<20; j++) {
			for(k=0; k<20; k++) {
				dp[0][j][k]+=(s[i]=='A'?k:j+1);
				dp[1][j][k]+=(s[i]=='A'?k+1:j);
			}
		}
	}
	cout<<min(dp[0][0][0],dp[1][0][0])<<'\n';
	return 0;
}
