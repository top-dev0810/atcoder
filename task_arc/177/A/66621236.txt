#include<bits/stdc++.h>
using namespace std;
#define endl '\n'
#define V vector
#define F first
#define S second
#define qp priority_queue<ll,vector<ll>,greater<ll> >
typedef long long ll;
using pii=pair<ll,ll>;
const int N=1e6+10;
ll T,n,a[N],b[N],c[N];
signed main(){
	ios::sync_with_stdio(0);
	cin.tie(0),cout.tie(0);
	c[1]=1,c[2]=5,c[3]=10,c[4]=50,c[5]=100,c[6]=500;
	for(int i=1;i<=6;i++)cin>>b[i];
	cin>>n;
	for(int i=1;i<=n;i++){
		cin>>a[i];
		for(int j=6;j>=1&&a[i]>0;j--){
			if((a[i]/c[j])<=b[j]){
				b[j]-=a[i]/c[j];
				a[i]-=a[i]/c[j]*c[j];
			}
			else{
				a[i]-=c[j]*b[j];
				b[j]=0;
			}
		}
		if(a[i]>0){
			cout<<"No"<<endl;
			return 0;
		}
	}
	cout<<"Yes"<<endl;
	return 0;
}
