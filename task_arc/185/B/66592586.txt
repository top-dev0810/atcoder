    #define rep(i,n) for (int i = 0; i < (n); ++i)

    typedef long long ll;
    typedef unsigned long long ull;
    using P=pair<int,int>;
    using T=tuple<ll,ll,ll>;
    const int INF=1001001001;
    const ll INFL=1e18;
    const ll mod=998244353;

    void solve(){
        int n;
        cin>>n;
        vector<ll>a(n);
        rep(i,n)cin>>a[i];
        ll sum=accumulate(a.begin(),a.end(),0ll);
        ll m=sum/n,r=sum%n,rem=0;
        for(int i=n-1;i>=0;i--){
            ll x=m;
            if(i>=n-r)x++;
            if(a[i]>=x)rem+=a[i]-x;
            else{
                if(a[i]+rem<x){
                    cout<<"No"<<endl;
                    return;
                }
                rem-=x-a[i];
                a[i]=x;
            }
        }
        cout<<"Yes"<<endl;
	}
    
    int main(){
        ios_base::sync_with_stdio(false);
        cin.tie(NULL);
		
		int t;
		cin>>t;
		rep(i,t){
			solve();
		}
        
        return 0;
    }
