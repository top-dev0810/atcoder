const int maxn=2e5;
const int mod=998244353;
int ksm(int x,int y){
	if(!y)return 1;
	int fw=ksm(x,y>>1);
	if(y&1)return fw*fw%mod*x%mod;
	return fw*fw%mod;
}
int n,m,inv[maxn+5];
signed main(){
	cin>>n>>m;
	for(int i=1;i<=maxn;i++)inv[i]=ksm(i,mod-2);
	int ans=((n-1)*m*m+n*m*(m-1))%mod;
	for(int i=0;i<n;i++)(ans+=m*(i*(2*m-1)+n)%mod*inv[n-i])%=mod;
	cout<<ans;
	return 0;
}
