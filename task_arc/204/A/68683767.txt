#include<bits/stdc++.h>
using namespace std;
const int mod = 998244353;
int f[5010][5010];
int g[5010];
int a[5010],b[5010];
int fuck(int n,int x){
    f[0][0] = 1;
    for(int i = 1;i<=n;++i){
        f[0][i] = 1;
        if(a[i]-b[0]-a[n]+b[n]>x){
            f[0][i] = 0;
        }
    }
    for(int i = 1;i<=n;++i){
        f[i][i] = f[i-1][i];
        for(int j = i+1;j<=n;++j){
            f[i][j] = f[i][j-1]+f[i-1][j];
            f[i][j]%=mod;
        }
        for(int j = i;j<=n;++j){
            if(a[j]-b[i]-a[n]+b[n]>x){
                f[i][j] = 0;
            }
        }
    }
    return f[n][n];
}
int main(){
    int n,l,r;
    cin>>n>>l>>r;
    for(int i = 1;i<=n;++i){
        cin>>a[i];
        a[i]+=a[i-1];
        a[i]%=mod;
    }
    for(int i = 1;i<=n;++i){
        cin>>b[i];
        b[i] += b[i - 1];
        b[i] %= mod;
    }
    cout<<(fuck(n,r)-fuck(n,l-1)+mod)%mod;
}
