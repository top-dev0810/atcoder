#include<bits/stdc++.h>
using namespace std;
int main() {
  int l, r, n;
  cin >> n >> l >> r;
  int A = l;
  int B = r - l;
  int C = n - A - B;
  //cout << A << " " << B << " " << C << endl;
  if (A > B) {
    cout << "Yes\n";
    while (C > 0) {
      cout << n - 1 << " ";
      C--;
      n--;
    }
    cout << B << " ";
    n--;
    while (n > B) {
      cout << n << " ";
      n--;
    }
  }
  else {
    vector<int> ans;
    while (C > 0) {
      int take = min(C - 1, A);
      ans.emplace_back(n - 1 - take);
      for (int z = 0; z < take; z++) {
        ans.emplace_back(n - 1 - z);
      }
      C--;
      A -= take;
      n -= (take + 1);
    }
    if (A != 0) cout << "No\n";
    else {
      assert(n == B);
      cout << "Yes\n";
      for (int x : ans) cout << x << " ";
      cout << '\n';
    }
  }
}
