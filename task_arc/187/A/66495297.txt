#include <bits/stdc++.h>
using namespace std;
using ll = long long;
using ld = long double;
using P = pair<int, int>;
const ll mod = 998244353;

int main() {
    int n, k;
    cin >> n >> k;

    vector<int> a(n);
    for (int i=0; i<n; i++) cin >> a[i];
    
    if (n==2) {
      int d = a[0]-a[1];
      if (d<=0) cout << "Yes" << endl << 0 << endl;
      else if (0<d && d<k) cout << "No" << endl;
      else cout << "Yes" << endl << 1 << endl << 1 << endl;
      return 0;
    }
    cout << "Yes" << endl;
    
    vector<int> ans;
    int d = a[0]-a[1];
    while (a[0]>a[1]) {
      ans.push_back(2);
      ans.push_back(2);
      a[1] += k;
      a[2] += k;
    }
    
    for (int i=2; i<n; i++) {
      int d = a[i-1]-a[i];
      if (d<=0) {
        continue;
      } else if (0<d && d<k) {
        while (a[i-2]>a[i]) {
          ans.push_back(i);
          ans.push_back(i);
          a[i] += k;
          a[i-1] += k;
        }
        ans.push_back(i-1);
        ans.push_back(i-1);
        ans.push_back(i);
        swap(a[i-1], a[i]);
        a[i-2] += k;
        a[i-1] += k;
        a[i] += k;
      } else {
        swap(a[i-1], a[i]);
        a[i-1] += k;
        ans.push_back(i);
        
        while (a[i-2]>a[i-1]) {
          ans.push_back(i);
          ans.push_back(i);
          a[i-1] += k;
          a[i] += k;
        }
      }
    }
    
    cout << ans.size() << endl;
    for (int i : ans) cout << i << " "; cout << endl;

    return 0;
}
