#include<bits/stdc++.h>
#define int long long
#define endl '\n'

using namespace std;

int n,T,A[6],P[6];

signed main(){
	ios::sync_with_stdio(0);cin.tie(0);cout.tie(0);cin>>T;
	while(T--){
		for(int i=1;i<=5;++i) cin>>A[i];
		for(int i=1;i<=5;++i) cin>>P[i];
		int nd=A[1]*2+A[2]-A[4]-A[5]*2;
		if(nd<=0) cout<<0<<endl;
		else if(P[4]*2<=P[5]) cout<<nd*P[4]<<endl;
		else cout<<((nd/2*P[5])+((nd&1)?min(P[4],P[5]):0))<<endl;
	}
	return 0;
}
