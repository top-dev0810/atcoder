#include <bits/stdc++.h>
using namespace std;
#define int long long int
const int N = 2e5 + 5;
int t, n;
double a[N], suma[N];
main() {
    ios::sync_with_stdio(false), cin.tie(0);
    // freopen("***.in", "r", stdin);
    // freopen("***.out", "w", stdout);
    for (cin >> t; t; t--) {
        cin >> n;
        int sum = 0;
        for (int i = 1; i <= n; i++) cin >> a[i];
        sort(a + 1, a + n + 1);
        for (int i = 1;i <= n; i++) suma[i] = suma[i - 1] + a[i];
        int ans = 0;
        for (int i = 1; i <= n; i++) {
            int idx = upper_bound(a + 1, a + i + 1, suma[i] / i) - a;
            ans = max(ans, i - idx + 1);
        }
        cout << ans << '\n';
    }
    return 0;
}
