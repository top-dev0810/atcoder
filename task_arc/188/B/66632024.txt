void solve()
{
    int n, k;
    cin >> n >> k;
    if (n % 2 == 1)
    {
        if (__gcd(n, 2 * k) == 1)
        {
            cout << "Yes" << endl;
            return;
        }
        else
        {
            cout << "No" << endl;
            return;
        }
    }
    else
    {
        if (__gcd(n, 2 * k) == 1)
        {
            cout << "Yes" << endl;
            return;
        }
        if (__gcd(n, 2 * k) == 2 && (n / 2) % 2 == 1)
        {
            cout << "Yes" << endl;
            return;
        }
        cout << "No" << endl;
        return;
    }
}
