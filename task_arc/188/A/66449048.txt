int qpow(int n, int k)
{
    int res = 1;
    while (k)
    {
        if (k & 1)
            (res *= n) %= MOD;
        (n *= n) %= MOD;
        k >>= 1;
    }
    return res;
}

map<array<int, 5>, int> dp[100];

void work()
{
    int n, m;
    cin >> n >> m;
    string s;
    cin >> s;
    s = s;
    dp[0][{0, 0, 0, 0, 0}] = 1;
    for (int i = 0; i < n; i++)
    {
        for (auto x : dp[i])
        {
            if (s[i] != 'B' && s[i] != 'C')
            {
                (dp[i + 1][{x.first[3] + 1, x.first[2], x.first[1], x.first[0], x.first[4] + x.first[0]}] += x.second) %= MOD;
            }
            if (s[i] != 'A' && s[i] != 'C')
            {
                (dp[i + 1][{x.first[2], x.first[3] + 1, x.first[0], x.first[1], x.first[4] + x.first[1]}] += x.second) %= MOD;
            }
            if (s[i] != 'A' && s[i] != 'B')
            {
                (dp[i + 1][{x.first[1], x.first[0], x.first[3] + 1, x.first[2], x.first[4] + x.first[2]}] += x.second) %= MOD;
            }
        }
    }
    int ans = 0;
    for (auto x : dp[n])
    {
        if (x.first[4] >= m)
            ans += x.second;
    }
    cout << ans % MOD << endl;
}
