int qpow(int n, int k)
{
    int res = 1;
    while (k)
    {
        if (k & 1)
            (res *= n) %= MOD;
        (n *= n) %= MOD;
        k >>= 1;
    }
    return res;
}

int L[100],R[100];

bool qry(int x,int y)
{
    cout<<"? "<<x<<' '<<y<<endl;
    cin>>x;
    return x;
}

vector<int>ans;

vector<int>tmp1,tmp0;

void work()
{
    int n, m, q;
    cin >> n >> m >> q;
    const int SIZ=19;
    int cnt=n/SIZ;
    for(int i=1;i<=cnt;i++)
    {
        L[i]=R[i-1]+1;
        R[i]=L[i]+SIZ-1;
    }
    R[cnt]=n;
    vector<int>vec1,vec0;
    vec0.push_back(L[cnt]);
    for(int i=L[cnt]+1;i<=R[cnt];i++)
    {
        if(qry(L[cnt],i))vec1.push_back(i);
        else vec0.push_back(i);
    }
    if(vec0.size()<vec1.size())
    {
        for(auto x:vec0)ans.push_back(x);
    }
    else
    {
        for(auto x:vec1)ans.push_back(x);
    }
    for(int i=1;i<cnt;i++)
    {
        vec1.clear();
        vec0.clear();
        vec0.push_back(L[i]);
        for(int j=L[i]+1;j<=R[i];j++)
        {
            if(qry(L[i],j))vec1.push_back(j);
            else vec0.push_back(j);
        }
        if(vec1.size()==9||vec0.size()==9)
        {
            tmp1=vec1;
            tmp0=vec0;
        }
        else
        {
            if(vec0.size()<vec1.size())
            {
                for(auto x:vec0)ans.push_back(x);
            }
            else
            {
                for(auto x:vec1)ans.push_back(x);
            }
        }
    }
    if(tmp0.size())
    {
        if(ans.size())
        {
            if(tmp0.size()==9)for(auto x:tmp0)ans.push_back(x);
            else for(auto x:tmp1)ans.push_back(x);
        }
        else
        {
            if(tmp0.size()==10)for(auto x:tmp0)ans.push_back(x);
            else for(auto x:tmp1)ans.push_back(x);
        }
    }
    cout<<"! ";
    sort(ans.begin(),ans.end());
    for(auto x:ans)cout<<x<<' ';
    cout<<endl;
}

    {
