#include <bits/stdc++.h>
using namespace std;


void burn() {
    long long x, y, z;
    cin >> x >> y >> z;
    if (2 * x < y)
        return cout << "No\n", void();
    if ((y & 1) && !z)
        return cout << "No\n", void();
    if ((y & 1))
        z--;

    z -= min(z, y / 2);
    if (!z)
        return cout << "Yes\n", void();

    long long rem = x - (y + 1) / 2;
    if (rem >= z)
        cout << "Yes\n";
    else
        cout << "No\n";
}

int main() {
    ios_base::sync_with_stdio(false), cin.tie(nullptr), cout.tie(nullptr);
    int t = 1;
    cin >> t;
    while (t--)
        burn();
}
