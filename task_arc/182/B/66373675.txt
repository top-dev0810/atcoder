int n, m;

void dfs(int n, int m, int x) {
    if (n == 1) {
        cout << (x << m) << ' ';
        return;
    }
    if (m == 0) {
        for (; n; -- n) {
            cout << x << ' ';
        }
        return;
    }
    dfs(    n / 2, m - 1, x << 1    );
    dfs(n - n / 2, m - 1, x << 1 | 1);
}

void INIT() { }

void WORK() {
    cin >> n >> m;
    dfs(n, m - 1, 1);
    cout << '\n';
}
