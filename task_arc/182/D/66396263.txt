void read(vector<T> &vec)
{

int qpow(int n,int k)
{
    int res=1;
    while(k)
    {
        if(k&1)(res*=n)%=MOD;
        (n*=n)%=MOD;
        k>>=1;
    }
    return res;
}

int n,m;

int calc(int tmp,int d,vector<int>&res)
{
    if(tmp==0)
    {
        tmp=d;
    }
    else if(tmp<0)
    {
        tmp=(tmp/m-1)*m+d;
    }
    else 
    {
        tmp=tmp/m*m+d;
    }
    int ans=0;
    for(auto x:res)ans+=abs(x-tmp);
    return ans;
}

void work()
{
    cin>>n>>m;
    vector<int>a(n),b(n);
    read(a);
    read(b);
    if(m==2)
    {
        if(a[0]!=b[0])cout<<-1<<endl;
        else cout<<0<<endl;
        return ;
    }
    vector<int>c(n);
    c[0]=b[0];
    for(int i=1;i<n;i++)
    {
        c[i]=c[i-1]+(b[i]-b[i-1]+m)%m;
        if(a[i]<a[i-1])c[i]-=m;
    }
    for(int i=0;i<n;i++)c[i]=a[i]-c[i];
    // for(auto x:c)cout<<x<<' ';
    sort(c.begin(),c.end());
    cout<<min(calc(c[n/2],0,c),calc(c[n/2],m,c))<<endl;
}
