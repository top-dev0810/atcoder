int qpow(int n,int k)
{
    int res=1;
    while(k)
    {
        if(k&1)(res*=n)%=MOD;
        (n*=n)%=MOD;
        k>>=1;
    }
    return res;
}

int p[10010],v[10010];

int f[10010][2];

void work()
{
    int n,q;
    cin>>n>>q;
    for(int i=1;i<=q;i++)
    {
        cin>>p[i]>>v[i];
        f[i][0]=f[i][1]=1;
        for(int j=1;j<i;j++)
        {
            if(v[i]>=v[j])continue;
            if(p[i]==p[j])
            {
                cout<<0<<endl;
                return ;
            }
            if(p[i]<p[j])
            {
                f[i][0]=f[j][1]=0;
            }
            else
            {
                f[i][1]=f[j][0]=0;
            }
        }
    }
    int ans=1;
    for(int i=1;i<=q;i++)ans=ans*(f[i][0]+f[i][1])%MOD;
    cout<<ans<<endl;
}

    {
        work();
