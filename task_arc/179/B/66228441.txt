#include <bits/stdc++.h>
using namespace std;
using ll = long long;
const ll md = 998244353;
ll n,m,mm[10005][2030];
vector<ll> g[22];
void init()
{
    cin.tie(0);
    cout.tie(0);
    ios_base::sync_with_stdio(0);
}
ll dp(ll vt, ll ch)
{
    if(vt > n) return 1;
    if(mm[vt][ch] != -1) return mm[vt][ch];

    bitset<10> s(ch);
    ll res = 0;
    for(ll i = 0; i < m; i++)
    {
        if(s[i]) continue;
        s[i] = 1;
        bitset<10> q(s);

        for(auto v:g[i]) q[v] = 0;
        res += dp(vt + 1, q.to_ullong());
        res %= md;
        s[i] = 0;
    }

    return mm[vt][ch] = res;
}
int main()
{
    init();
    cin >> m >> n;
    memset(mm, -1, sizeof mm);
    for(ll i = 0; i < m; i++)
    {
        ll a;
        cin >> a;
        g[a - 1].push_back(i);
    }

    cout << dp(1,0);
}
